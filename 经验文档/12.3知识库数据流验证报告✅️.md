def _evaluate_knowledge_consistency(self, content: str) -> DimensionScore:
    """V2.1新增：评估知识库一致性"""
    try:
        recall = _get_knowledge_recall()
        
        if recall is None:
            # 知识库不可用，返回默认评分
            return DimensionScore(
                dimension_name='知识库一致性',
                score=0.8,  # 知识库不可用时给默认分
                details='知识库模块未启用',
                issues=[],
                knowledge_conflicts=None,
                recalled_knowledge=None
            )

        check_result = recall.check_knowledge_consistency(
            content=content,
            category=None,  # 自动识别题材
            top_k=10
        )
        
        # 提取冲突信息
        conflicts = []
        if check_result.conflicts:
            for conflict in check_result.conflicts:
                conflicts.append({
                    "type": conflict.conflict_type,
                    "severity": conflict.severity,
                    "description": conflict.description,
                    "knowledge_title": conflict.knowledge_title,
                    "suggested_fix": conflict.suggested_fix
                })
        
        # 提取召回的知识点
        recalled = []
        if check_result.recalled_knowledge:
            for knowledge in check_result.recalled_knowledge[:5]:
                recalled.append({
                    "title": knowledge.title,
                    "category": knowledge.category,
                    "domain": knowledge.domain,
                })
        
        # 计算评分
        # 一致性评分 = 1.0 - (P0冲突数 × 0.3 + P1冲突数 × 0.1 + P2冲突数 × 0.05)
        p0_count = sum(1 for c in conflicts if c['severity'] == 'P0')
        p1_count = sum(1 for c in conflicts if c['severity'] == 'P1')
        p2_count = sum(1 for c in conflicts if c['severity'] == 'P2')
        
        score = max(0.0, 1.0 - (p0_count * 0.3 + p1_count * 0.1 + p2_count * 0.05))
        
        return DimensionScore(
            dimension_name='知识库一致性',
            score=score,
            details=f'召回{len(recalled)}个知识点，检测到{len(conflicts)}个冲突',
            issues=[c['description'] for c in conflicts[:3]],
            knowledge_conflicts=conflicts,
            recalled_knowledge=recalled
        )

        generated_content, validation_fn
    )

### 4.3 提示词构建 ⚠️

**已实现的集成方案** (`core/generation_orchestrator.py`):

```python
class GenerationOrchestrator:
    """创作流程编排器 - 统一管理两种知识库的调用"""
    
    def prepare_generation(
        self,
        knowledge_bases: List[str],
        writing_techniques: List[str],
        workspace_root: Path
    ) -> Dict[str, Any]:
        """
        准备生成所需的上下文和提示词
        
        Args:
            knowledge_bases: 选中的传统知识库列表
            writing_techniques: 选中的写作技巧列表
            workspace_root: 工作区根目录
            
        Returns:
            包含完整提示词和配置的字典
        """
        result = {
            "traditional_knowledge": None,
            "writing_technique_rules": None,
            "combined_prompt": "",
            "metadata": {}
        }
        
        # 1. 处理传统知识库
        if knowledge_bases:
            selector = KnowledgeReferenceSelector(workspace_root)
            strategies = selector.select_strategies(knowledge_bases)
            result["traditional_knowledge"] = strategies
        
        # 2. 处理写作技巧库
        if writing_techniques:
            integrator = WritingTechniqueIntegrator(workspace_root)
            rules = integrator.build_mandatory_prompt(writing_techniques)
            result["writing_technique_rules"] = rules
            
            # 构建强制遵循提示词
            mandatory_prompt = f"""
【写作技巧强制规则】

{rules}

⚠️ 以上写作技巧规则必须100%遵守，违反将导致评分扣减。
"""
            result["combined_prompt"] += mandatory_prompt
        
        return result
```
