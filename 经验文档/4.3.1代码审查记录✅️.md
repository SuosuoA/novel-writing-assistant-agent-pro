```python
    with self._available:
        start_time = time.time()
        while True:
            # 查找空闲连接
            for conn_info in self._pool:
                # 检查连接是否超时被强制回收
                if conn_info.in_use:
                    elapsed = (datetime.now() - conn_info.last_used).total_seconds()
                    if elapsed > self._max_connection_time:  # 新增配置
                        logger.warning(f"连接使用超时，强制回收: {elapsed:.1f}s")
                        conn_info.in_use = False
                
                if not conn_info.in_use:
                    conn_info.in_use = True
                    conn_info.last_used = datetime.now()
                    return conn_info
            
            # ... 超时检查 ...

def close(self, timeout: float = 5.0) -> None:
    """优雅关闭连接池"""
    with self._lock:
        # 等待所有连接归还
        start_time = time.time()
        while any(c.in_use for c in self._pool):
            if time.time() - start_time > timeout:
                logger.warning(f"关闭超时，强制关闭 {sum(1 for c in self._pool if c.in_use)} 个使用中的连接")
                break
            time.sleep(0.1)
        
        for conn_info in self._pool:
            try:
                conn_info.connection.close()
            except Exception as e:
                logger.error(f"关闭连接失败: {e}")
        
        self._pool.clear()
        self._closed = True  # 新增关闭标记
```

```python
def _calculate_directory_hash(directory: Path) -> str:
    hasher = hashlib.sha256()
    
    for file_path in directory.rglob("*"):
        # 跳过__pycache__
        if file_path.name == "__pycache__":
            continue
        
        # 跳过.pyc文件
        if file_path.suffix == ".pyc":
            continue
        
        # 跳过目录
        if file_path.is_dir():
            continue
        
        try:
            with open(file_path, "rb") as f:
                hasher.update(f.read())
        except Exception:
            continue
    
    return hasher.hexdigest()
```

```python
@dataclass
class HistogramData:
    values: List[float] = field(default_factory=list)
    max_size: int = 10000  # 最大保留样本数
    
    def add(self, value: float) -> None:
        if len(self.values) >= self.max_size:
            # 滑动窗口：移除最老的10%
            self.values = self.values[self.max_size // 10:]
        
        self.values.append(value)
        self.count += 1
        self.sum += value
        self.min = min(self.min, value)
        self.max = max(self.max, value)
```
