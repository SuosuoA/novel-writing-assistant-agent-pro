## 六、API参考

### 6.1 学习闭环管理器 (LearningLoopManager)

```python
from core.learning_loop_manager import LearningLoopManager

# 初始化
manager = LearningLoopManager(workspace_root=Path("project/"))

# 提取知识点
knowledge_points = manager.extract_knowledge_points(
    content="章节内容...",
    chapter_id="chapter_001"
) -> List[KnowledgePoint]

# 收集生成数据
manager.collect_generation_data(
    chapter_id="chapter_001",
    content="章节内容...",
    scores={"word_count": 0.85, "style": 0.88}
)

# 更新知识库
manager.update_knowledge_base(knowledge_points)  # 单参数

# 获取统计
stats = manager.get_learning_stats() -> LearningStats
```

### 6.2 评分历史分析器 (ScoreHistoryAnalyzer)

```python
from core.score_history_analyzer import ScoreHistoryAnalyzer

# 初始化
analyzer = ScoreHistoryAnalyzer(workspace_root=Path("project/"))

# 记录评分
analyzer.record_score("word_count", 0.85, "chapter_001")

# 获取趋势
trend = analyzer.get_trend("word_count") -> str  # "stable"/"improving"/"declining"

# 获取平均分
avg = analyzer.get_average_score("word_count") -> float
```

### 6.3 冲突模式学习器 (ConflictPatternLearner)

```python
from core.conflict_pattern_learner import ConflictPatternLearner

# 初始化
learner = ConflictPatternLearner(workspace_root=Path("project/"))

# 学习修正模式
pattern_id = learner.learn_from_correction(
    original="张三身穿蓝色长袍",
    corrected="张三穿着一身天蓝色的丝绸长袍",
    conflict_type="style"  # 必需参数
) -> str

# 预测冲突
conflicts = learner.predict_conflicts(content="测试内容") -> List[Dict]

# 获取统计
stats = learner.get_pattern_stats() -> Dict[str, Any]
```

### 6.4 AI感检测器 (AIFeelingDetector)

```python
from core.ai_feeling_detector import AIFeelingDetector

# 初始化（无参数）
detector = AIFeelingDetector()

# 检测AI感
report = detector.detect("他不禁感到一阵悲伤") -> AIFeelingReport

# 获取AI感评分
score = detector.get_ai_feeling_score("文本内容") -> float

# 获取AI痕迹词汇
words = detector.get_ai_words("文本内容") -> List[str]
```

### 6.5 反馈收集器 (FeedbackCollector)

```python
from core.feedback_collector import FeedbackCollector

# 初始化（db_path参数）
collector = FeedbackCollector(db_path=Path("data/feedback.db"))

# 收集反馈
feedback = collector.collect(
    chapter_id="chapter_001",
    feedback_text="对话太生硬",
    feedback_type="style",  # content/style/ai_feeling/other
    context={"section": "dialogue"}
) -> Feedback

# 标记已处理
collector.mark_processed(feedback.id)  # 传入id，不是对象

# 获取统计
stats = collector.get_statistics() -> Dict
```

### 6.6 反馈提纯器 (FeedbackPurifier)

```python
from core.feedback_purifier import FeedbackPurifier

# 初始化（无参数）
purifier = FeedbackPurifier()

# 提纯反馈
knowledge = purifier.purify(
    feedback_text="对话太生硬",
    feedback_type="style"
) -> List[KnowledgePoint]
```

### 6.7 策略调整器 (StrategyAdjuster)

```python
from core.strategy_adjuster import StrategyAdjuster

# 初始化（db_path参数）
adjuster = StrategyAdjuster(db_path=Path("data/strategy.db"))

# 根据反馈调整
adjustment = adjuster.adjust_from_feedback(
    feedback_type="style",
    feedback_text="对话太生硬",
    extracted_knowledge=None  # 可选
) -> StrategyAdjustment

# 获取优化Prompt
prompt = adjuster.get_optimized_prompt("base_prompt") -> str
```
