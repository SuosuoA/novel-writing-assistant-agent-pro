    """获取AI感评分（兼容性别名）"""
    report = self.detect(text)
    return report.total_score

def get_ai_words(self, text: str) -> List[str]:
    """获取AI痕迹词汇列表（兼容性别名）"""
    report = self.detect(text)
    ai_words = []
