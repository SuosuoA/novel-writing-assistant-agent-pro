### P0-1: SQL注入风险 - DatabaseMigration.executescript()

**位置**: `core/database.py:200`

**问题**:
```python
def apply_migration(self, name: str) -> bool:
    sql = self._migrations[name]
    conn = self._pool.get_connection()
    # 执行迁移SQL
    conn.executescript(sql)  # 直接执行SQL脚本
```

**风险分析**:
- `executescript()` 允许执行多条SQL语句
- 如果迁移脚本来源不可信，可能导致SQL注入
- 迁移脚本注册时无验证机制

**攻击场景**:
1. 恶意插件注册包含恶意SQL的迁移脚本
2. 执行迁移时导致数据泄露或破坏

**建议修复**:
```python
def register_migration(self, name: str, sql: str) -> None:
    """注册迁移脚本 - 添加安全验证"""
    # 1. 验证迁移名称格式
    if not re.match(r'^\d{3}_[a-z_]+$', name):
        raise ValueError(f"Invalid migration name: {name}")
    
    # 2. 禁止危险操作（DROP/DELETE/TRUNCATE需要显式允许）
    dangerous_patterns = [
        r'DROP\s+TABLE',
        r'DELETE\s+FROM',
        r'TRUNCATE\s+TABLE',
        r'--.*$',  # SQL注释（可能隐藏恶意代码）
    ]
    for pattern in dangerous_patterns:
        if re.search(pattern, sql, re.IGNORECASE):
            raise ValueError(
                f"Migration {name} contains dangerous SQL pattern: {pattern}"
            )
    
    self._migrations[name] = sql

def apply_migration(self, name: str) -> bool:
    """应用迁移 - 添加事务保护"""
    if name not in self._migrations:
        logger.error(f"Migration not found: {name}")
        return False

    sql = self._migrations[name]
    conn = self._pool.get_connection()

    try:
        # 使用事务保护
        conn.execute("BEGIN TRANSACTION")
        
        # 拆分并逐条执行（而不是executescript）
        statements = [s.strip() for s in sql.split(';') if s.strip()]
        for stmt in statements:
            conn.execute(stmt)
        
        # 记录迁移
        conn.execute(
            """INSERT INTO _migrations (name, applied_at, status) VALUES (?, ?, ?)""",
            (name, datetime.now().isoformat(), MigrationStatus.APPLIED.value)
        )
        conn.commit()
        return True
    except Exception as e:
        conn.rollback()
        logger.error(f"Migration failed: {name}, error: {e}")
        return False
```

**位置**: `core/database.py:42-66`

**问题**:

- 高并发场景下可能创建大量连接
- 资源耗尽攻击(DoS)

**建议修复**:
```python
class ConnectionPool:
    def __init__(self, db_path: str, pool_size: int = 5, timeout: float = 30.0):
        self._db_path = db_path
        self._timeout = timeout
        self._max_pool_size = pool_size  # 使用池大小限制
        self._lock = threading.RLock()
        self._thread_local = threading.local()
        
        self._all_connections: List[sqlite3.Connection] = []
        self._connections_lock = threading.Lock()
        self._connection_count = 0  # 连接计数
        
    def get_connection(self) -> sqlite3.Connection:
        """获取连接 - 添加大小限制"""
        if (
            not hasattr(self._thread_local, "connection")
            or self._thread_local.connection is None
        ):
            # 检查连接池是否已满
            with self._connections_lock:
                if self._connection_count >= self._max_pool_size:
                    # 等待可用连接或超时
                    raise RuntimeError(
                        f"Connection pool exhausted (max: {self._max_pool_size})"
                    )
                self._connection_count += 1
            
            try:
                conn = sqlite3.connect(
                    self._db_path, timeout=self._timeout, check_same_thread=False
                )
                # ... 配置连接 ...
                
                self._thread_local.connection = conn
                
                with self._connections_lock:
                    self._all_connections.append(conn)
            except Exception:
                with self._connections_lock:
                    self._connection_count -= 1
                raise
        
        return self._thread_local.connection
```

### P1-2: 配置管理器无输入验证

**位置**: `core/config_manager.py:197-229`

**问题**:

**风险分析**:

- 可能注入恶意配置值
- 配置值类型不一致可能导致运行时错误

**建议修复**:
```python
class ConfigValidator:
    """配置验证器 - 添加内置验证规则"""
    
    # 内置验证规则
    BUILTIN_VALIDATORS = {
        # AI配置
        "ai.temperature": lambda v: isinstance(v, (int, float)) and 0.0 <= v <= 2.0,
        "ai.max_tokens": lambda v: isinstance(v, int) and 1 <= v <= 100000,
        "ai.timeout": lambda v: isinstance(v, int) and 1 <= v <= 300,
        
        # 插件配置
        "plugins.hot_reload_debounce": lambda v: isinstance(v, (int, float)) and 0.1 <= v <= 10.0,
        "plugins.enabled": lambda v: isinstance(v, bool),
        
        # 数据库配置
        "database.wal_mode": lambda v: isinstance(v, bool),
    }
    
    def __init__(self):
        self._validators: Dict[str, Callable[[Any], bool]] = {}
        # 注册内置验证器
        for key, validator in self.BUILTIN_VALIDATORS.items():
            self._validators[key] = validator
    
    def validate(self, key_path: str, value: Any) -> bool:
        """验证配置值"""
        # 优先使用自定义验证器
        if key_path in self._validators:
            return self._validators[key_path](value)
        
        # 内置验证器
        if key_path in self.BUILTIN_VALIDATORS:
            return self.BUILTIN_VALIDATORS[key_path](value)
        
        # 无验证器时，进行基本类型检查
        if value is None:
            return True
        
        # 检查是否是安全类型
        safe_types = (str, int, float, bool, list, dict)
        if not isinstance(value, safe_types):
            logger.warning(f"Unsafe config type for {key_path}: {type(value)}")
            return False
        
        return True
```

### P1-5: 缓存键碰撞风险

**风险分析**:

```python
def _cache_result(self, prompt: str, result: str) -> None:
    """缓存结果 - 使用更安全的哈希"""
    import hashlib
    
    # 使用SHA256替代MD5
    cache_key = hashlib.sha256(prompt.encode()).hexdigest()
    
    # 添加提示词长度作为额外标识（减少碰撞概率）
    cache_key = f"{cache_key}:{len(prompt)}"
    
    self._cache.set(cache_key, {
        "result": result,
        "timestamp": datetime.now(timezone.utc),
        "prompt_hash": hashlib.sha256(prompt.encode()).hexdigest()[:16],  # 用于验证
    })

def _get_cached_result(self, prompt: str) -> Optional[str]:
    """获取缓存结果 - 添加完整性验证"""
    import hashlib
    
    cache_key = hashlib.sha256(prompt.encode()).hexdigest()
    cache_key = f"{cache_key}:{len(prompt)}"
    
    cached = self._cache.get(cache_key)
    
    if cached is None:
        return None
    
    # 验证提示词哈希（防止碰撞）
    expected_hash = hashlib.sha256(prompt.encode()).hexdigest()[:16]
    if cached.get("prompt_hash") != expected_hash:
        logger.warning("Cache collision detected, invalidating cache entry")
        return None
    
    age = (datetime.now(timezone.utc) - cached["timestamp"]).total_seconds()
    
    if age > self._cache_ttl:
        return None
    
    return cached["result"]
```

```python
class DeadLetterQueue:
    """死信队列 - 添加自动清理机制"""
    
    def __init__(self, max_size: int = 100, ttl: int = 86400):

        self._ttl = ttl
        self._lock = threading.Lock()
    
    def add(self, event: Event, error: Exception) -> None:
        """添加失败事件 - 自动清理过期条目"""
        with self._lock:
            # 清理过期条目
            current_time = time.time()
            self._queue = [
                entry for entry in self._queue
                if current_time - entry.get("timestamp", 0) < self._ttl
            ]
            
            # 检查队列大小
            if len(self._queue) >= self._max_size:
                self._queue.pop(0)  # 移除最旧的
            
            # 脱敏敏感数据
            from infrastructure.security import mask_sensitive
            event_data = mask_sensitive(event.model_dump())
            
            self._queue.append({
                "event": event_data,
                "error": str(error),  # 避免存储异常对象
                "timestamp": current_time,
            })
    
    def get_all(self) -> List[Dict[str, Any]]:
        """获取所有死信 - 自动清理过期条目"""
        with self._lock:
            current_time = time.time()
            self._queue = [
                entry for entry in self._queue
                if current_time - entry.get("timestamp", 0) < self._ttl
            ]
            return list(self._queue)
    
    def clear(self) -> None:
        """清空死信队列"""
        with self._lock:
            self._queue.clear()
```
