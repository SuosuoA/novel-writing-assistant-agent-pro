#### 问题定位

```python
# plugins/hot-ranking-v1/plugin.py 添加缓存方法

import json
import time
from pathlib import Path

class HotRankingPlugin(ToolPlugin):
    CACHE_FILE = "hot_ranking_cache.json"
    CACHE_EXPIRE_HOURS = 24
    
    def _get_cached_data(self) -> Optional[Dict]:
        """获取缓存数据"""
        cache_path = Path(self.cache_file)
        if cache_path.exists():
            try:
                with open(cache_path, 'r', encoding='utf-8') as f:
                    cache = json.load(f)
                    # 检查缓存是否过期
                    cache_time = cache.get('timestamp', 0)
                    if time.time() - cache_time < self.CACHE_EXPIRE_HOURS * 3600:
                        return cache.get('data')
            except Exception as e:
                logger.warning(f"读取热榜缓存失败: {e}")
        return None
    
    def _save_cache(self, data: Dict):
        """保存缓存数据"""
        try:
            cache_path = Path(self.cache_file)
            cache = {
                'timestamp': time.time(),
                'data': data
            }
            with open(cache_path, 'w', encoding='utf-8') as f:
                json.dump(cache, f, ensure_ascii=False, indent=2)
        except Exception as e:
            logger.warning(f"保存热榜缓存失败: {e}")
```

```python
# gui_main.py 修改 _create_hot_ranking_page 方法

def _create_hot_ranking_page(self) -> tk.Frame:
    """创建热榜页面（异步加载版本）"""
    frame = ttk.Frame(self._content_frame, style="TFrame")
    
    # ... UI布局代码 ...
    
    # ===== 关键修改：先显示缓存数据，异步爬取新数据 =====
    
    # 1. 先显示缓存数据（或默认数据）
    cached_data = self._get_cached_hot_ranking_data()
    self._display_hot_ranking_data(cached_data)
    
    # 2. 显示"正在更新..."提示
    self._hot_ranking_progress_var.set("正在后台更新数据...")
    self._hot_ranking_progress_frame.pack(fill=tk.X, padx=20, pady=5)
    
    # 3. 异步爬取新数据
    self._async_fetch_hot_ranking()
    
    return frame

def _get_cached_hot_ranking_data(self) -> Dict:
    """获取缓存的热榜数据"""
    cache_file = os.path.join(os.path.dirname(__file__), "hot_ranking_cache.json")
    if os.path.exists(cache_file):
        try:
            with open(cache_file, 'r', encoding='utf-8') as f:
                cache = json.load(f)
                # 检查缓存是否过期（24小时）
                cache_time = cache.get('timestamp', 0)
                if time.time() - cache_time < 24 * 3600:
                    logger.info("[热榜] 使用缓存数据")
                    return cache.get('data')
        except Exception as e:
            logger.warning(f"读取热榜缓存失败: {e}")
    
    # 返回默认数据
    return self._get_default_hot_ranking_data()

def _async_fetch_hot_ranking(self):
    """异步爬取热榜数据"""
    def fetch_task():
        try:
            # 调用热榜插件爬取数据
            if CORE_AVAILABLE:
                registry = get_plugin_registry()
                if registry:
                    hot_ranking_plugin = registry.get_plugin("hot-ranking-v1")
                    if hot_ranking_plugin:
                        result = hot_ranking_plugin.execute("get_data", {"force_fresh": True})
                        # 保存缓存
                        self._save_hot_ranking_cache(result)
                        # 回调主线程更新UI
                        self.root.after(0, lambda: self._on_hot_ranking_fetched(result))
                        return
        except Exception as e:
            logger.error(f"热榜爬取失败: {e}")
            self.root.after(0, lambda: self._on_hot_ranking_fetch_failed(str(e)))
    
    # 使用线程池提交任务
    from core.thread_pool_manager import thread_pool_manager
    thread_pool_manager.submit_sync(fetch_task)

def _on_hot_ranking_fetched(self, data: Dict):
    """热榜数据获取成功回调（主线程执行）"""
    self._display_hot_ranking_data(data)
    self._hot_ranking_progress_var.set("数据已更新")
    self._hot_ranking_progress_bar['value'] = 100
    # 3秒后隐藏进度条
    self.root.after(3000, lambda: self._hot_ranking_progress_frame.pack_forget())

def _on_hot_ranking_fetch_failed(self, error: str):
    """热榜数据获取失败回调（主线程执行）"""
    self._hot_ranking_progress_var.set(f"更新失败: {error}")
    logger.error(f"热榜更新失败: {error}")
```

### 3.3 向量模型延迟加载方案

#### 方案选择

| 方案 | 描述 | 优点 | 缺点 | 推荐度 |
|------|------|------|------|--------|
| 方案1 | 延迟加载（懒加载） | 启动快，按需加载 | 首次使用有延迟 | ⭐⭐⭐⭐⭐ |
| 方案2 | 后台预加载 | 启动快，无首次延迟 | 实现稍复杂 | ⭐⭐⭐⭐ |
| 方案3 | 启动画面+同步加载 | 实现简单 | 用户等待 | ⭐⭐⭐ |

**推荐方案：方案1 + 方案2组合**

#### 实施代码

**步骤1：修改VectorStore为延迟加载**

```python
# infrastructure/vector_store.py

class VectorStore:
    """向量存储（延迟加载版本）"""
    
    _instance = None
    _model = None
    _model_loaded = False
    _loading = False
    _lock = threading.Lock()
    
    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
        return cls._instance
    
    def __init__(self):
        # 不在初始化时加载模型！
        self.model_name = "all-MiniLM-L6-v2"
        self.cache_dir = os.path.join("F:\\", "sentence_transformers_cache")
        self._model_loaded = False
    
    def _ensure_model_loaded(self):
        """延迟加载模型（首次调用时才加载）"""
        if self._model_loaded:
            return
        
        with self._lock:
            if self._model_loaded:
                return
            
            if self._loading:
                # 等待其他线程加载完成
                while self._loading:
                    time.sleep(0.1)
                return
            
            self._loading = True
            try:
                logger.info("[VectorStore] 开始加载向量模型...")
                from sentence_transformers import SentenceTransformer
                
                # 确保离线模式
                os.environ["HF_HUB_OFFLINE"] = "1"
                
                # 加载模型
                self._model = SentenceTransformer(
                    self.model_name, 
                    cache_folder=self.cache_dir
                )
                self._model_loaded = True
                logger.info(f"[VectorStore] 向量模型加载完成: {self.model_name}")
            except Exception as e:
                logger.error(f"[VectorStore] 向量模型加载失败: {e}")
                raise
            finally:
                self._loading = False
    
    def embed(self, text: str) -> np.ndarray:
        """编码文本（延迟加载模型）"""
        self._ensure_model_loaded()
        return self._model.encode(text)
    
    def embed_batch(self, texts: List[str]) -> np.ndarray:
        """批量编码文本（延迟加载模型）"""
        self._ensure_model_loaded()
        return self._model.encode(texts)
    
    def is_model_loaded(self) -> bool:
        """检查模型是否已加载"""
        return self._model_loaded
```
