# 9.6.4 参考文本解析与融合实现说明

## 1. 概述

### 1.1 设计目标

参考文本解析与融合模块旨在让用户能够提供参考文本（如已有小说、世界观设定、人物设定等），系统自动从中提取关键元素并融合到生成Prompt中，使生成内容更加贴合用户的创作风格和设定。

### 1.2 核心功能

| 功能 | 描述 |
|------|------|
| 文本类型检测 | 自动识别参考文本类型（小说/大纲/人物设定/世界观设定） |
| 世界观提取 | 提取时代背景、力量体系、势力组织等元素 |
| 人物提取 | 提取姓名、性格、外貌、能力等人物设定 |
| 情节提取 | 提取关键事件、冲突、转折点等情节元素 |
| 风格识别 | 识别文本风格关键词（热血、爽文、治愈等） |
| Prompt融合 | 将解析结果智能融合到生成Prompt中 |

### 3.2 世界观元素提取

使用正则表达式匹配关键信息：

```python
def _extract_worldview(self, text: str) -> WorldviewElements:
    elements = WorldviewElements()
    
    # 提取时代背景
    for kw in self.WORLDVIEW_KEYWORDS["era"]:
        pattern = rf"({kw}[：:是为]?\s*([^\n。！？]+))"
        match = re.search(pattern, text)
        if match:
            elements.era = match.group(2).strip()
            break
    
    # 提取力量等级（支持多种格式）
    for pattern in self.POWER_LEVEL_PATTERNS:
        matches = re.findall(pattern, text)
        elements.power_levels.extend(matches)
    
    # ...其他元素提取
    return elements
```

### 3.3 人物元素提取

支持结构化和非结构化文本：

```python
def _extract_character(self, text: str) -> CharacterElements:
    elements = CharacterElements()
    
    # 提取姓名（多种模式）
    name_patterns = [
        r"姓名[：:]\s*([^\n，。]+)",
        r"名字[：:]\s*([^\n，。]+)",
        r"([^\n，。]{2,4})[是为]?(主角|男主|女主)"
    ]
    
    # 提取能力（支持逗号分隔）
    abilities_text = match.group(2).strip()
    elements.abilities.extend(re.split(r"[，、,]", abilities_text))
    
    return elements
```

### 3.4 情节元素提取

```python
def _extract_plot(self, text: str) -> PlotElements:
    elements = PlotElements()
    
    # 提取关键事件
    event_patterns = [
        r"([^\n。！？]*(发生|遭遇|经历)[^\n。！？]*)",
        r"事件[：:]\s*([^\n。]+)"
    ]
    
    # 提取冲突
    for kw in self.PLOT_KEYWORDS["conflicts"]:
        pattern = rf"([^\n。！？]*{kw}[^\n。！？]*)"
        matches = re.findall(pattern, text)
        elements.conflicts.extend(matches)
    
    return elements
```

## 4. 融合器实现

### 4.1 世界观Prompt融合

```python
@staticmethod
def fusion_to_worldview_prompt(parsed: ParsedReference, variables: Dict) -> Dict:
    fused = variables.copy()
    
    if not parsed.has_worldview():
        return fused
    
    wv = parsed.worldview
    reference_context = []
    
    if wv.era:
        reference_context.append(f"参考时代背景：{wv.era}")
    if wv.power_system:
        reference_context.append(f"参考力量体系：{wv.power_system}")
    if wv.power_levels:
        reference_context.append(f"参考等级体系：{'、'.join(wv.power_levels[:5])}")
    
    if reference_context:
        fused["reference_text"] = "\n".join(reference_context)
    
    return fused
```

### 4.2 人物Prompt融合

```python
@staticmethod
def fusion_to_character_prompt(parsed: ParsedReference, variables: Dict) -> Dict:
    fused = variables.copy()
    
    if not parsed.has_characters():
        return fused
    
    reference_context = []
    
    for i, char in enumerate(parsed.characters[:5]):
        char_info = []
        if char.name:
            char_info.append(f"姓名：{char.name}")
        if char.role:
            char_info.append(f"定位：{char.role}")
        if char.personality:
            char_info.append(f"性格：{'、'.join(char.personality[:3])}")
        
        if char_info:
            reference_context.append(f"参考人物{i+1}：" + "，".join(char_info))
    
    if reference_context:
        fused["reference_text"] = "\n".join(reference_context)
    
    return fused
```

### 4.3 大纲Prompt融合

支持多类型元素融合：

```python
@staticmethod
def fusion_to_outline_prompt(parsed: ParsedReference, variables: Dict) -> Dict:
    fused = variables.copy()
    reference_context = []
    
    # 融合世界观参考
    if parsed.has_worldview():
        wv = parsed.worldview
        if wv.era or wv.power_system:
            reference_context.append(f"参考世界观：{wv.era}，{wv.power_system}")
    
    # 融合情节参考
    if parsed.has_plot():
        plot = parsed.plot
        if plot.events:
            events_text = "、".join(e.get("description", "")[:20] for e in plot.events[:3])
            reference_context.append(f"参考关键事件：{events_text}")
    
    # 融合人物参考
    if parsed.has_characters():
        char_names = [c.name for c in parsed.characters[:5] if c.name]
        if char_names:
            reference_context.append(f"参考人物：{'、'.join(char_names)}")
    
    return fused
```

### 5.3 新增公开API

```python
def parse_reference_text(self, text: str) -> Dict[str, Any]:
    """
    解析参考文本并返回解析结果（V1.2新增）
    
    用于预览参考文本中提取的元素，帮助用户了解
    系统从参考文本中识别了哪些世界观、人物、情节元素。
    """
    if not text or not self._reference_parser:
        return {"error": "参考文本为空或解析器未初始化"}
    
    parsed = self._reference_parser.parse(text)
    
    return {
        "reference_type": parsed.reference_type.value,
        "confidence": parsed.confidence,
        "worldview": parsed.worldview.to_dict() if parsed.has_worldview() else None,
        "characters": [c.to_dict() for c in parsed.characters] if parsed.has_characters() else [],
        "plot": parsed.plot.to_dict() if parsed.has_plot() else None,
        "style_keywords": parsed.style_keywords,
        "summary": {
            "has_worldview": parsed.has_worldview(),
            "has_characters": parsed.has_characters(),
            "has_plot": parsed.has_plot(),
            "character_count": len(parsed.characters),
            "style_count": len(parsed.style_keywords)
        }
    }
```

```python
def parse(self, text: str, reference_type: str = None) -> ParsedReference:
    # 检查缓存
    cache_key = self._get_cache_key(text)
    if cache_key in self._cache:
        return self._cache[cache_key]
    
    # 解析并缓存
    result = ParsedReference(...)
    self._cache[cache_key] = result
    return result
```
