## 三、去重机制

### 3.1 三层去重策略

```python
class DedupManager:
    """三层去重管理器"""
    
    def check_duplicate(self, title: str, content: str, category: str) -> bool:
        """
        三层去重检查
        
        Layer 1: 标题完全匹配（快速）
        Layer 2: SimHash指纹相似度（中等）
        Layer 3: 向量语义相似度（深度）
        """
        # Layer 1: 标题检查
        if self._check_title_dup(title, category):
            return True
        
        # Layer 2: SimHash指纹
        simhash = self._compute_simhash(content)
        if self._check_simhash_dup(simhash, threshold=0.95):
            return True
        
        # Layer 3: 向量相似度
        if self._check_vector_dup(content, threshold=0.90):
            return True
        
        return False
```

### 3.2 SimHash算法

```python
def _compute_simhash(self, text: str) -> int:
    """计算SimHash指纹"""
    import jieba
    
    # 分词
    words = jieba.cut(text)
    
    # 计算词频
    word_freq = {}
    for word in words:
        word_freq[word] = word_freq.get(word, 0) + 1
    
    # 计算哈希向量
    v = [0] * 64
    for word, freq in word_freq.items():
        word_hash = hash(word)
        for i in range(64):
            if word_hash & (1 << i):
                v[i] += freq
            else:
                v[i] -= freq
    
    # 生成指纹
    fingerprint = 0
    for i in range(64):
        if v[i] > 0:
            fingerprint |= (1 << i)
    
    return fingerprint

def _hamming_distance(self, h1: int, h2: int) -> int:
    """计算汉明距离"""
    return bin(h1 ^ h2).count('1')

def _similarity(self, h1: int, h2: int) -> float:
    """计算相似度（0-1）"""
    return 1.0 - self._hamming_distance(h1, h2) / 64.0
```

## 四、并行生成优化

### 4.1 动态批处理

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
import time

class ParallelGenerator:
    """并行生成器"""
    
    def __init__(self, max_workers: int = 5):
        self.max_workers = max_workers
        self.executor = ThreadPoolExecutor(max_workers=max_workers)
        self.progress_tracker = ProgressTracker()
    
    def generate_batch(self, topics: List[str], category: str) -> List[Dict]:
        """
        并行生成一批知识点
        
        优化策略：
        1. 动态批次大小（根据API响应时间调整）
        2. 指数退避重试
        3. 失败任务重新入队
        """
        results = []
        batch_size = self._calculate_batch_size()
        
        # 分批处理
        for i in range(0, len(topics), batch_size):
            batch = topics[i:i+batch_size]
            
            # 提交任务
            futures = {
                self.executor.submit(self._generate_one, topic, category): topic
                for topic in batch
            }
            
            # 收集结果
            for future in as_completed(futures):
                topic = futures[future]
                try:
                    result = future.result(timeout=60)
                    results.append(result)
                    self.progress_tracker.mark_success(topic)
                except Exception as e:
                    logger.error(f"生成失败 {topic}: {e}")
                    # 重试机制
                    self._retry_later(topic, category)
        
        return results
    
    def _calculate_batch_size(self) -> int:
        """动态计算批次大小"""
        # 根据API响应时间动态调整
        avg_response_time = self.progress_tracker.get_avg_response_time()
        
        if avg_response_time < 3:
            return 10  # 快速响应，大批次
        elif avg_response_time < 6:
            return 5   # 中等响应，中批次
        else:
            return 3   # 慢速响应，小批次
```

### 4.2 指数退避重试

```python
def _retry_later(self, topic: str, category: str, retry_count: int = 0):
    """指数退避重试"""
    max_retries = 3
    
    if retry_count >= max_retries:
        logger.error(f"超过最大重试次数，放弃: {topic}")
        return
    
    # 指数退避延迟
    delay = 2 ** retry_count  # 1s, 2s, 4s
    
    time.sleep(delay)
    
    try:
        result = self._generate_one(topic, category)
        logger.info(f"重试成功: {topic}")
        return result
    except Exception as e:
        logger.warning(f"重试失败 ({retry_count+1}/{max_retries}): {topic}")
        return self._retry_later(topic, category, retry_count + 1)
```

## 五、质量控制

### 5.1 六项质量检查

```python
class QualityChecker:
    """质量检查器"""
    
    def check_knowledge(self, knowledge: Dict) -> Tuple[bool, List[str]]:
        """
        六项质量检查
        
        返回：(是否通过, 问题列表)
        """
        issues = []
        
        # 1. 字数检查（至少2000字）
        word_count = len(knowledge.get('content', ''))
        if word_count < 2000:
            issues.append(f"字数不足：{word_count} < 2000")
        
        # 2. 关键词检查（至少5个）
        keywords = knowledge.get('keywords', [])
        if len(keywords) < 5:
            issues.append(f"关键词不足：{len(keywords)} < 5")
        
        # 3. 案例检查（至少2个）
        cases = knowledge.get('cases', [])
        if len(cases) < 2:
            issues.append(f"案例不足：{len(cases)} < 2")
        
        # 4. 参考文献检查（至少2个）
        refs = knowledge.get('references', [])
        if len(refs) < 2:
            issues.append(f"参考文献不足：{len(refs)} < 2")
        
        # 5. 写作应用检查（至少3个建议）
        applications = knowledge.get('writing_applications', [])
        if len(applications) < 3:
            issues.append(f"写作应用不足：{len(applications)} < 3")
        
        # 6. AI痕迹检查
        ai_patterns = ['首先', '其次', '最后', '总之', '综上所述']
        content = knowledge.get('content', '')
        ai_count = sum(1 for p in ai_patterns if p in content)
        if ai_count > 3:
            issues.append(f"AI痕迹过多：{ai_count}个AI模式词")
        
        return len(issues) == 0, issues
```

## 六、进度追踪与断点续传

### 6.1 进度追踪器

```python
class ProgressTracker:
    """进度追踪器"""
    
    def __init__(self, save_path: str = "data/knowledge_progress.json"):
        self.save_path = save_path
        self.progress = self._load_progress()
    
    def mark_success(self, topic: str):
        """标记成功"""
        self.progress['success'].append(topic)
        self.progress['last_update'] = time.time()
        self._save_progress()
    
    def mark_failed(self, topic: str, error: str):
        """标记失败"""
        self.progress['failed'].append({
            'topic': topic,
            'error': error,
            'timestamp': time.time()
        })
        self._save_progress()
    
    def get_pending_topics(self, all_topics: List[str]) -> List[str]:
        """获取待处理主题（去重）"""
        completed = set(self.progress['success'])
        return [t for t in all_topics if t not in completed]
```

### 6.2 断点续传

```python
def resume_generation(self):
    """
    断点续传
    
    从上次中断的位置继续生成
    """
    # 加载进度
    tracker = ProgressTracker()
    
    # 获取所有待生成主题
    all_topics = self._get_all_topics()
    
    # 排除已完成的
    pending_topics = tracker.get_pending_topics(all_topics)
    
    logger.info(f"发现 {len(pending_topics)} 个待生成主题")
    
    # 继续生成
    self.generate_batch(pending_topics, self.category)
```

### 7.2 缓存机制

```python
from functools import lru_cache
import hashlib

class CachedGenerator:
    """带缓存的生成器"""
    
    @lru_cache(maxsize=100)
    def _get_cached_response(self, prompt_hash: str) -> Dict:
        """缓存响应结果"""
        # 实际生成逻辑
        pass
    
    def generate(self, topic: str, category: str) -> Dict:
        """生成知识点（带缓存）"""
        # 计算提示词哈希
        prompt = f"{category}_{topic}"
        prompt_hash = hashlib.md5(prompt.encode()).hexdigest()
        
        # 尝试从缓存获取
        cached = self._get_cached_response(prompt_hash)
        if cached:
            logger.info(f"命中缓存: {topic}")
            return cached
        
        # 实际生成
        result = self._actual_generate(topic, category)
        
        # 存入缓存
        self._get_cached_response(prompt_hash)
        
        return result
```
