## 五、BaseAgent抽象基类

### 5.1 类定义

```python
class BaseAgent(ABC):
    """Agent抽象基类"""
    
    def __init__(self, agent_type: str, metadata: AgentMetadata = None):
        self.agent_type = agent_type
        self._metadata = metadata or self._create_default_metadata()
        self._status = AgentStatus(agent_type=agent_type, state=AgentState.UNLOADED)
        self._initialized = False
        self._service_locator = None  # ServiceLocator实例
        self._logger: Optional[logging.Logger] = None
    
    # === 抽象方法 ===
    
    @abstractmethod
    def initialize(self) -> bool: ...
    
    @abstractmethod
    def execute(self, task_id: str, payload: Dict[str, Any], 
                context: AgentContext = None) -> AgentResult: ...
    
    @abstractmethod
    def can_handle(self, task_type: str, payload: Dict[str, Any]) -> bool: ...
    
    @abstractmethod
    def cleanup(self) -> bool: ...
    
    # === ServiceLocator集成 ===
    
    def set_service_locator(self, service_locator) -> None: ...
    def get_service(self, service_type: Type) -> Any: ...
    def get_config(self) -> Dict[str, Any]: ...
    def get_logger(self) -> logging.Logger: ...
    def get_event_bus(self): ...
```

### 5.2 使用示例

```python
class MyAgent(BaseAgent):
    def __init__(self):
        metadata = AgentMetadata(
            agent_type="my_agent",
            name="My Agent",
            capabilities=[AgentCapability.GENERATION],
        )
        super().__init__("my_agent", metadata)
    
    def initialize(self) -> bool:
        self._set_state(AgentState.ACTIVE)
        self._initialized = True
        return True
    
    def execute(self, task_id: str, payload: Dict[str, Any], 
                context: AgentContext = None) -> AgentResult:
        # 通过ServiceLocator获取服务
        config = self.get_config()
        logger = self.get_logger()
        event_bus = self.get_event_bus()
        
        # 执行任务逻辑
        result_data = {"processed": True, "input": payload}
        
        return AgentResult.success_result(
            task_id=task_id,
            agent_type=self.agent_type,
            data=result_data,
        )
    
    def can_handle(self, task_type: str, payload: Dict[str, Any]) -> bool:
        return task_type == "my_task"
    
    def cleanup(self) -> bool:
        self._initialized = False
        return True
```

### 6.1 核心组件

```
MasterAgent
├── AgentRegistry      # Agent注册表

├── DependencyResolver # 依赖解析器
├── ThreadPoolExecutor # 线程池执行器

```

### 6.2 任务调度流程

```
1. 提交任务 → submit_task()
2. 添加到TaskQueue
3. 注册依赖到DependencyResolver
4. 调度循环检测就绪任务
5. 查找能处理的Agent
6. 异步执行任务
7. 记录结果/发布事件
```

### 6.3 主要方法

```python
class MasterAgent:
    # === Agent管理 ===
    def register_agent(self, agent: BaseAgent) -> bool
    def unregister_agent(self, agent_type: str) -> bool
    def get_agent(self, agent_type: str) -> Optional[BaseAgent]
    def list_agents(self) -> List[str]
    def find_agents_by_capability(self, capability: AgentCapability) -> List[str]
    
    # === 任务调度 ===
    def start(self) -> None
    def stop(self) -> None
    def submit_task(self, task_type: str, payload: Dict, 
                   priority: TaskPriority = TaskPriority.NORMAL,
                   dependencies: List[str] = None) -> str
    def cancel_task(self, task_id: str) -> bool
    def get_task_result(self, task_id: str) -> Optional[AgentResult]
    def get_task_status(self, task_id: str) -> str
    
    # === 流水线执行 ===
    def register_pipeline(self, pipeline: PipelineDefinition) -> None
    def execute_pipeline(self, pipeline_name: str, payload: Dict) -> Dict[str, AgentResult]
    
    # === ServiceLocator集成 ===
    def set_service_locator(self, service_locator) -> None
    def get_service(self, service_type: Type) -> Any
    def get_event_bus(self)
```

### 6.4 流水线定义

```python
@dataclass
class PipelineStage:
    name: str                              # 阶段名称
    agent_type: str                        # Agent类型
    dependencies: List[str] = field(default_factory=list)  # 依赖阶段
    timeout_seconds: int = 30              # 超时时间
    optional: bool = False                 # 可选阶段

@dataclass
class PipelineDefinition:
    name: str                              # 流水线名称
    stages: List[PipelineStage]            # 阶段列表
    description: str = ""                  # 描述
```

### 6.5 使用示例

```python
# 创建MasterAgent
master = MasterAgent(max_workers=10)

# 设置ServiceLocator
master.set_service_locator(service_locator)

# 注册Agent
agent1 = MyAgent("agent1")
agent2 = MyAgent("agent2")
master.register_agent(agent1)
master.register_agent(agent2)

# 定义流水线
pipeline = PipelineDefinition(
    name="generation_pipeline",
    stages=[
        PipelineStage(name="plan", agent_type="planner"),
        PipelineStage(name="generate", agent_type="generator", dependencies=["plan"]),
        PipelineStage(name="validate", agent_type="validator", dependencies=["generate"]),
    ],
)
master.register_pipeline(pipeline)

# 启动调度器
master.start()

# 执行流水线
results = master.execute_pipeline(
    pipeline_name="generation_pipeline",
    payload={"input": "写一段描述春天的文字"},
)

# 停止调度器
master.stop()
```

class TaskPriority(IntEnum):

### 9.2 循环依赖检测

**使用DFS（深度优先搜索）算法检测循环依赖**：

```python
def detect_cycle(self) -> Optional[List[str]]:
    """使用DFS检测循环依赖"""
    WHITE, GRAY, BLACK = 0, 1, 2
    color = {task_id: WHITE for task_id in self._dependency_graph}
    
    def dfs(task_id: str) -> Optional[List[str]]:
        color[task_id] = GRAY
        for dep_id in self._dependency_graph[task_id]:
            if color.get(dep_id, WHITE) == GRAY:
                # 发现环（GRAY表示正在访问中）
                return [..., dep_id]  # 返回环路径
            elif color.get(dep_id, WHITE) == WHITE:
                result = dfs(dep_id)
                if result:
                    return result
        color[task_id] = BLACK
        return None
```

**注意**: 文档早期版本描述使用Kahn算法，实际实现使用DFS算法。DFS更适合检测环路径，Kahn适合拓扑排序。

## 十二、最佳实践

### 12.1 Agent实现建议

1. **始终提供Metadata**: 明确定义Agent的能力和约束
2. **使用Context传递状态**: 避免全局变量
3. **返回标准化Result**: 便于结果处理和监控
4. **实现can_handle**: 精确匹配任务类型

### 12.2 MasterAgent使用建议

1. **先注册Agent再启动**: 确保所有Agent已初始化
2. **设置ServiceLocator**: Agent才能访问核心服务
3. **使用流水线**: 复杂任务分解为多个阶段
4. **监控任务状态**: 定期检查get_stats()
