## 工作流程概览

```
步骤1: 整理打包请求内容
    SmartContextBuilder构建优化的提示词
    包含：世界观设定、人物设定、风格档案、大纲内容、上下文记忆
    ⚠️ 强制要求：章节结束必须添加【本章完】标记
    ↓
步骤2: 向大模型发送请求
    IterativeGeneratorV2发送API请求
    支持本地模型和在线API（通过OpenAI标准接口）
    生成策略：CREATIVE/BALANCED/PRECISE
    ↓
步骤3: 接受返回文章
    接收AI生成的内容
    ↓
步骤4: 从多维度评分
    EnhancedWeightedValidator进行6维度加权评分:
    • 字数符合性（权重10%）
    • 大纲符合性（权重15%）
    • 风格一致性（权重25%）
    • 人设一致性（权重25%）
    • 世界观一致性（权重20%）
    • 自然度（权重5%）
    ⚠️ 没有章节结束标记【本章完】视作未完成
    ↓
步骤5: 判断是否达标
    总评分 < 0.8 → 构建反馈，循环优化
    总评分 ≥ 0.8 且有【本章完】 → 输出保存
    最多迭代5次
    ↓
步骤6: 构建反馈（如果未达标）
    总评分 + 各维度评分
    具体修改建议（针对各维度）
    重新生成优化
    ↓
步骤7: 输出保存
    达标后保存最终内容
    记录评分历史和迭代次数
```

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
增强版加权评分验证器
实现用户要求的6维度加权评分系统：
1. 字数符合性（权重：10%）
2. 大纲符合性（权重：25%）
3. 风格一致性（权重：25%）
4. 人设一致性（权重：15%）
5. 世界观一致性（权重：20%，严重违背时一票否决）
6. 自然度（权重：5%）
"""

import re
import logging
from typing import Dict, List, Any, Optional, Tuple
from dataclasses import dataclass
from collections import Counter


@dataclass
class WordCountScore:
    """字数符合性评分"""
    target_words: int
    actual_words: int
    difference: int
    accuracy_percentage: float
    score: float  # 0.4-1.0


@dataclass
class OutlineComplianceScore:
    """大纲符合性评分"""
    score: float  # 0.4-1.0
    matched_plot_points: int  # 匹配的情节点数量
    total_plot_points: int  # 总情节点数量
    matched_keywords: List[str]  # 匹配的关键词
    missing_plot_points: List[str]  # 未包含的情节点


@dataclass
class NaturalnessScore:
    """自然度评分（AI痕迹检测）"""
    score: float  # 0.4-1.0
    ai_probability: float  # AI生成概率
    formulaic_score: float  # 公式化程度
    cliche_score: float  # 陈词滥调程度
    issues_found: List[str]


@dataclass
class WeightedValidationResult:
    """加权评分验证结果"""
    # 各维度评分
    word_count_score: WordCountScore
    outline_compliance: OutlineComplianceScore
    style_consistency: float  # 使用原有的风格一致性评分
    character_consistency: float  # 人设一致性
    worldview_consistency: float  # 世界观一致性
    naturalness: NaturalnessScore
    
    # 加权总分
    total_weighted_score: float  # 0.0-1.0
    
    # 是否通过验证
    passed: bool
    
    # 详细反馈
    feedback: Dict[str, Any]
    
    # 改进建议
    suggestions: List[str]


class EnhancedWeightedValidator:
    """增强版加权评分验证器"""
    
    # 评分权重配置(优化后 - 强调设定优先级)
    WEIGHTS = {
        'word_count': 0.10,      # 字数符合性 10%
        'outline': 0.15,         # 大纲符合性 15%(Prompt中已强化,权重降低)
        'style': 0.25,           # 风格一致性 25%(权重最高)
        'character': 0.25,      # 人设一致性 25%(提升权重,与风格并列最高)
        'worldview': 0.20,      # 世界观一致性 20%(权重高,严重违背一票否决)
        'naturalness': 0.05      # 自然度 5%(权重低)
    }
    
    def __init__(self, logger: logging.Logger = None):
        """
        初始化验证器
        
        Args:
            logger: 日志记录器
        """
        self.logger = logger or logging.getLogger(__name__)
        
        # AI痕迹检测模式
        self.ai_patterns = {
            "formulaic_beginnings": [
                r'在[\u4e00-\u9fff]{2,6}的[\u4e00-\u9fff]{2,6}，',
                r'这是一个关于[\u4e00-\u9fff]{2,10}的故事',
                r'让我们把目光投向[\u4e00-\u9fff]{2,10}',
                r'话说[\u4e00-\u9fff]{2,10}',
                r'在遥远的[\u4e00-\u9fff]{2,10}'
            ],
            "over_explanation": [
                r'换句话说，',
                r'也就是说，',
                r'具体来说，',
                r'详细地说，',
                r'简而言之，'
            ],
            "repetitive_structures": [
                r'一方面，[\u4e00-\u9fff]+。另一方面，[\u4e00-\u9fff]+。',
                r'首先，[\u4e00-\u9fff]+。其次，[\u4e00-\u9fff]+。最后，[\u4e00-\u9fff]+。',
                r'不仅[\u4e00-\u9fff]+，而且[\u4e00-\u9fff]+'
            ],
            "unnatural_transitions": [
                r'突然，',
                r'就在这时，',
                r'没想到，',
                r'出乎意料的是，',
                r'令人惊讶的是，'
            ],
            "ai_cliches": [
                r'在.*的背景下',
                r'从.*的角度来看',
                r'值得注意的是',
                r'需要强调的是',
                r'不可否认的是'
            ]
        }
        
        # 陈词滥调词汇
        self.cliches = [
            '美丽', '漂亮', '高大', '细小', '明亮', '黑暗',
            '重要', '关键', '核心', '基本', '主要', '必要',
            '显然', '明显', '清楚', '明白', '明确', '确定',
            '非常', '极其', '十分', '特别', '相当', '颇为',
            '大概', '可能', '也许', '或许', '或者', '恐怕',
            '事实上', '实际上', '其实', '实质上', '本质上'
        ]
    
    def validate_with_weights(
        self,
        text: str,
        target_word_count: int,
        chapter_outline: str = None,
        style_profile: Dict[str, Any] = None,
        character_profiles: List[Dict] = None,
        world_view: str = None
    ) -> WeightedValidationResult:
        """
        使用加权评分系统验证文本
        
        Args:
            text: 待验证的文本
            target_word_count: 目标字数
            chapter_outline: 章节大纲（用于大纲符合性检测）
            style_profile: 风格配置（用于风格一致性检测）
            character_profiles: 人物设定列表（用于人设一致性检测）
            world_view: 世界观设定（用于世界观一致性检测）
        
        Returns:
            加权评分验证结果
        """
        self.logger.info("开始加权评分验证...")
        
        # 0. 检查章节结束标记
        has_ending_marker = "【本章完】" in text
        if not has_ending_marker:
            self.logger.warning("⚠️  警告：缺少【本章完】结束标记，视为未完成")
        
        # 1. 字数符合性评分（会自动检查并记录结束标记）
        word_count_score = self._score_word_count(text, target_word_count)
        
        # 2. 大纲符合性评分
        if chapter_outline:
            outline_compliance = self._score_outline_compliance(text, chapter_outline)
        else:
            outline_compliance = OutlineComplianceScore(
                score=0.7,  # 没有大纲时给中等分数
                matched_plot_points=0,
                total_plot_points=0,
                matched_keywords=[],
                missing_plot_points=[]
            )
        
        # 3. 风格一致性评分（使用原有逻辑）
        style_consistency = self._score_style_consistency(text, style_profile)
        
        # 4. 人设一致性评分
        if character_profiles:
            character_consistency = self._score_character_consistency(text, character_profiles)
        else:
            character_consistency = 0.7  # 没有人设时给中等分数
        
        # 5. 世界观一致性评分（一票否决）
        if world_view:
            worldview_consistency, worldview_violation = self._score_worldview_consistency(text, world_view)
        else:
            worldview_consistency = 0.7
            worldview_violation = False
        
        # 6. 自然度评分
        naturalness = self._score_naturalness(text)
        
        # 检查严重违背世界观（一票否决）
        if worldview_violation:
            self.logger.warning("检测到严重违背世界观，一票否决")
            total_score = 0.0
            passed = False
        else:
            # 计算加权总分
            total_score = (
                word_count_score.score * self.WEIGHTS['word_count'] +
                outline_compliance.score * self.WEIGHTS['outline'] +
                style_consistency * self.WEIGHTS['style'] +
                character_consistency * self.WEIGHTS['character'] +
                worldview_consistency * self.WEIGHTS['worldview'] +
                naturalness.score * self.WEIGHTS['naturalness']
            )
            
            # 检查是否通过（必须同时满足：总分达标 + 包含结束标记）
            passed = (total_score >= 0.8 and has_ending_marker)  # 阈值0.8 + 必须有【本章完】
            
            if not has_ending_marker:
                self.logger.warning("⚠️  未包含【本章完】标记，判定为未完成")
        
        # 构建详细反馈
        feedback = {
            '章节结束标记': {
                'score': 1.0 if has_ending_marker else 0.0,
                'weight': 0.0,  # 不计入总分，但作为必要条件
                'weighted_score': 0.0,
                'details': '✓ 包含【本章完】' if has_ending_marker else '✗ 缺少【本章完】（必须添加）'
            },
            '字数符合性': {
                'score': word_count_score.score,
                'weight': self.WEIGHTS['word_count'],
                'weighted_score': word_count_score.score * self.WEIGHTS['word_count'],
                'details': f"目标{target_word_count}字，实际{word_count_score.actual_words}字"
            },
            '大纲符合性': {
                'score': outline_compliance.score,
                'weight': self.WEIGHTS['outline'],
                'weighted_score': outline_compliance.score * self.WEIGHTS['outline'],
                'details': f"匹配{outline_compliance.matched_plot_points}/{outline_compliance.total_plot_points}个情节点"
            },
            '风格一致性': {
                'score': style_consistency,
                'weight': self.WEIGHTS['style'],
                'weighted_score': style_consistency * self.WEIGHTS['style'],
                'details': '与学习风格的匹配度'
            },
            '人设一致性': {
                'score': character_consistency,
                'weight': self.WEIGHTS['character'],
                'weighted_score': character_consistency * self.WEIGHTS['character'],
                'details': '人物行为是否符合设定'
            },
            '世界观一致性': {
                'score': worldview_consistency,
                'weight': self.WEIGHTS['worldview'],
                'weighted_score': worldview_consistency * self.WEIGHTS['worldview'],
                'details': '是否符合世界观设定' + ('（一票否决）' if worldview_violation else '')
            },
            '自然度': {
                'score': naturalness.score,
                'weight': self.WEIGHTS['naturalness'],
                'weighted_score': naturalness.score * self.WEIGHTS['naturalness'],
                'details': f"AI痕迹概率: {naturalness.ai_probability:.1%}"
            }
        }
        
        # 生成改进建议
        suggestions = self._generate_weighted_suggestions(
            word_count_score, outline_compliance, style_consistency,
            character_consistency, worldview_consistency, naturalness
        )
        
        # 如果缺少结束标记，将其添加到建议列表的最前面
        if not has_ending_marker:
            suggestions.insert(0, "缺少【本章完】结束标记，请在章节末尾添加【本章完】")
        
        result = WeightedValidationResult(
            word_count_score=word_count_score,
            outline_compliance=outline_compliance,
            style_consistency=style_consistency,
            character_consistency=character_consistency,
            worldview_consistency=worldview_consistency,
            naturalness=naturalness,
            total_weighted_score=total_score,
            passed=passed,
            feedback=feedback,
            suggestions=suggestions
        )
        
        self.logger.info(f"加权评分完成，总分: {total_score:.2f}")
        return result
    
    def _score_word_count(self, text: str, target_words: int) -> WordCountScore:
        """字数符合性评分 - V5.8优化版
        
        改进点：
        1. 更严格的超标惩罚：超标50%以上给0.3分
        2. 超标100%以上给0.2分（触发重试）
        3. 区分超标和不足的不同容忍度
        """
        # 检查是否包含【本章完】标记
        has_ending_marker = "【本章完】" in text
        self.logger.info(f"检查章节结束标记: {'✓ 包含【本章完】' if has_ending_marker else '✗ 缺少【本章完】'}")
        
        # 统计字数（排除【本章完】这4个字）
        actual_words = self._count_words_excluding_marker(text)
        difference = actual_words - target_words
        accuracy_percentage = (actual_words / target_words * 100) if target_words > 0 else 100.0
        
        # V5.8优化：更严格的评分逻辑
        tolerance_10 = target_words * 0.10
        tolerance_20 = target_words * 0.20
        tolerance_30 = target_words * 0.30
        
        # 计算超标比例
        over_ratio = actual_words / target_words if target_words > 0 else 1.0
        
        if difference < 0:  # 字数不足
            if abs(difference) <= tolerance_10:
                score = 1.0
            elif abs(difference) <= tolerance_20:
                score = 0.8
            elif abs(difference) <= tolerance_30:
                score = 0.6
            else:
                score = 0.5  # 不足时相对宽容
        else:  # 字数超标
            if difference <= tolerance_10:
                score = 1.0
            elif difference <= tolerance_20:
                score = 0.75  # 超标惩罚更重
            elif difference <= tolerance_30:
                score = 0.5
            elif over_ratio <= 1.5:  # 超标30%-50%
                score = 0.3
            elif over_ratio <= 2.0:  # 超标50%-100%
                score = 0.2
                self.logger.warning(f"字数超标严重：{actual_words}字 vs 目标{target_words}字（超标{(over_ratio-1)*100:.0f}%）")
            else:  # 超标100%以上
                score = 0.1
                self.logger.error(f"字数超标超过100%！{actual_words}字 vs 目标{target_words}字")
        
        return WordCountScore(
            target_words=target_words,
            actual_words=actual_words,
            difference=difference,
            accuracy_percentage=accuracy_percentage,
            score=score
        )
    
    def _score_outline_compliance(self, text: str, chapter_outline: str) -> OutlineComplianceScore:
        """大纲符合性评分（权重：25%）"""
        # 提取情节点
        plot_points = self._extract_plot_points(chapter_outline)
        total_plot_points = len(plot_points)
        
        if total_plot_points == 0:
            return OutlineComplianceScore(
                score=0.7,
                matched_plot_points=0,
                total_plot_points=0,
                matched_keywords=[],
                missing_plot_points=[]
            )
        
        # 提取关键词
        keywords = self._extract_keywords(chapter_outline)
        
        # 检查文本中是否包含情节点
        matched_plot_points = 0
        missing_plot_points = []
        matched_keywords = []
        
        for plot_point in plot_points:
            # 简化判断：检查关键词是否出现
            keywords_in_plot = self._extract_keywords(plot_point)
            for keyword in keywords_in_plot:
                if keyword in text:
                    matched_plot_points += 1
                    break
            else:
                missing_plot_points.append(plot_point)
        
        # 检查关键词匹配
        for keyword in keywords:
            if keyword in text:
                matched_keywords.append(keyword)
        
        # 计算分数
        if total_plot_points > 0:
            match_ratio = matched_plot_points / total_plot_points
            if match_ratio >= 0.9:
                score = 1.0  # 完全符合
            elif match_ratio >= 0.7:
                score = 0.8  # 基本符合
            elif match_ratio >= 0.5:
                score = 0.6  # 部分符合
            else:
                score = 0.4  # 不符合
        else:
            score = 0.7
        
        return OutlineComplianceScore(
            score=score,
            matched_plot_points=matched_plot_points,
            total_plot_points=total_plot_points,
            matched_keywords=matched_keywords,
            missing_plot_points=missing_plot_points
        )
    
    def _score_style_consistency(self, text: str, style_profile: Dict[str, Any]) -> float:
        """风格一致性评分（权重：25%）- V5.5增强版
        
        改进内容：
        1. 支持多种风格数据格式（vocabulary_depth/vocabulary_profile）
        2. 从文本中提取写作风格特征进行多维度评分
        3. 检测AI痕迹、公式化表达、陈词滥调等
        4. 当style_profile为None或无效时，使用文本质量评分
        5. 基础分保底，避免过低评分
        """
        # 如果没有风格配置或配置无效，使用文本质量评分
        if not style_profile or not isinstance(style_profile, dict):
            score = self._score_text_quality(text)
            self.logger.info(f"[V5.5] 无风格配置，使用文本质量评分: {score:.2f}")
            return score
        
        import jieba  # 延迟导入
        words = list(jieba.cut(text))
        word_counts = Counter(words)
        
        scores = []
        
        # === 1. 词汇匹配度评分（权重40%）===
        # 支持多种格式
        style_words = []
        
        # 格式1: vocabulary_depth.high_frequency_words
        vocab_depth = style_profile.get('vocabulary_depth', {})
        if vocab_depth and 'high_frequency_words' in vocab_depth:
            high_freq = vocab_depth['high_frequency_words']
            for item in high_freq[:20]:
                if isinstance(item, list) and len(item) >= 1:
                    style_words.append(item[0])
                elif isinstance(item, str):
                    style_words.append(item)
        
        # 格式2: vocabulary_profile.most_common_words
        if not style_words:
            vocab_profile = style_profile.get('vocabulary_profile', {})
            if vocab_profile and 'most_common_words' in vocab_profile:
                for item in vocab_profile['most_common_words'][:20]:
                    if isinstance(item, list) and len(item) >= 1:
                        style_words.append(item[0])
                    elif isinstance(item, str):
                        style_words.append(item)
        
        # 格式3: style_tags 或 writing_characteristics
        if not style_words:
            style_tags = style_profile.get('style_tags', [])
            if style_tags:
                style_words.extend(style_tags[:10])
        
        # 过滤无效词汇
        style_words = [w for w in style_words if w and len(w) >= 2 and not w.startswith('#') and not w.startswith('*')]
        
        if style_words:
            text_common_words = [word for word, _ in word_counts.most_common(30)]
            matched_words = set(style_words).intersection(set(text_common_words))
            vocab_match_ratio = len(matched_words) / len(style_words) if style_words else 0
            
            # V5.5改进：更合理的评分逻辑
            if vocab_match_ratio >= 0.3:
                scores.append(1.0)
            elif vocab_match_ratio >= 0.2:
                scores.append(0.9)
            elif vocab_match_ratio >= 0.1:
                scores.append(0.8)
            elif vocab_match_ratio >= 0.05:
                scores.append(0.7)
            else:
                scores.append(0.6)  # 基础分保底
            
            self.logger.debug(f"[V5.5] 词汇匹配: {len(matched_words)}/{len(style_words)} = {vocab_match_ratio:.2%}")
        else:
            scores.append(self._score_text_quality(text))
        
        # === 2. 句式模式评分（权重30%）===
        sentence_patterns = style_profile.get('sentence_patterns', {})
        
        if sentence_patterns:
            sentences = self._split_sentences(text)
            if sentences:
                avg_length = sum(len(s) for s in sentences) / len(sentences)
                
                target_length = 30
                if 'avg_length' in sentence_patterns:
                    target_length = sentence_patterns['avg_length']
                elif 'avg_chars_per_sentence' in sentence_patterns:
                    target_length = sentence_patterns['avg_chars_per_sentence']
                elif 'sentence_length' in sentence_patterns:
                    target_length = sentence_patterns['sentence_length']
                
                if target_length > 0:
                    length_ratio = min(avg_length, target_length) / max(avg_length, target_length)
                    if length_ratio >= 0.7:
                        scores.append(1.0)
                    elif length_ratio >= 0.5:
                        scores.append(0.85)
                    elif length_ratio >= 0.3:
                        scores.append(0.7)
                    else:
                        scores.append(0.6)
                else:
                    scores.append(0.7)
            else:
                scores.append(0.7)
        else:
            scores.append(0.7)
        
        # === 3. 文本质量评分（权重30%）===
        quality_score = self._score_text_quality(text)
        scores.append(quality_score)
        
        # 加权平均
        weights = [0.4, 0.3, 0.3]
        final_score = sum(s * w for s, w in zip(scores, weights))
        
        # 确保最低分不低于0.6
        final_score = max(0.6, final_score)
        
        self.logger.info(f"[V5.5] 风格评分: 词汇{scores[0]:.2f}, 句式{scores[1]:.2f}, 质量{scores[2]:.2f} → 总分{final_score:.2f}")
        
        return final_score
    
    def _score_text_quality(self, text: str) -> float:
        """文本质量评分（无风格配置时的fallback）"""
        scores = []
        
        # 1. 句式多样性（权重30%）
        sentences = self._split_sentences(text)
        if sentences:
            sentence_lengths = [len(s) for s in sentences]
            avg_length = sum(sentence_lengths) / len(sentence_lengths)
            
            if 20 <= avg_length <= 40:
                scores.append(1.0)
            elif 15 <= avg_length <= 50:
                scores.append(0.8)
            else:
                scores.append(0.6)
        else:
            scores.append(0.5)
        
        # 2. AI痕迹检测（权重30%）
        ai_penalty = 0.0
        ai_patterns = [
            (r'在.*的.*下', 0.05),
            (r'从.*的.*角度', 0.05),
            (r'通过.*的.*方式', 0.05),
            (r'让我们', 0.03),
            (r'值得注意的是', 0.05),
            (r'需要强调的是', 0.05),
            (r'不可否认的是', 0.05),
            (r'事实上，', 0.03),
            (r'实际上，', 0.03),
            (r'突然，', 0.02),
            (r'就在这时，', 0.02),
        ]
        
        for pattern, penalty in ai_patterns:
            if re.search(pattern, text):
                ai_penalty += penalty
        
        ai_score = max(0.4, 1.0 - ai_penalty)
        scores.append(ai_score)
        
        # 3. 描写丰富度（权重25%）
        description_keywords = ['看着', '想', '感觉', '听到', '看到', '感受到', '意识到', '明白', '理解']
        description_count = sum(1 for kw in description_keywords if kw in text)
        
        ideal_count = len(text) / 500 * 3
        if description_count >= ideal_count * 0.8:
            scores.append(1.0)
        elif description_count >= ideal_count * 0.5:
            scores.append(0.8)
        elif description_count > 0:
            scores.append(0.6)
        else:
            scores.append(0.4)
        
        # 4. 语言自然度（权重15%）
        dialog_markers = ['"', '"', '「', '」', '『', '』']
        has_dialog = any(marker in text for marker in dialog_markers)
        
        detail_patterns = [r'\d+年', r'\d+月', r'\d+日', r'\d+点', r'红色', r'蓝色', r'黑色', r'白色']
        has_details = any(re.search(p, text) for p in detail_patterns)
        
        if has_dialog and has_details:
            scores.append(1.0)
        elif has_dialog or has_details:
            scores.append(0.8)
        else:
            scores.append(0.6)
        
        # 加权平均
        weights = [0.30, 0.30, 0.25, 0.15]
        final_score = sum(s * w for s, w in zip(scores, weights))
        
        return final_score
    
    def _score_character_consistency(self, text: str, character_profiles: List[Dict]) -> float:
        """人设一致性评分（权重：25%）- V5.5完全重写版
        
        核心改进：
        1. 智能提取性格关键词（支持markdown格式）
        2. 行为特征匹配：从文本中检测人物行为是否符合设定
        3. 对话风格匹配：检查人物对话是否符合设定
        4. 基础评分保底：即使无法提取性格关键词，也基于人物存在和提及频率给分
        """
        if not character_profiles:
            return 0.7
        
        consistency_scores = []
        character_issues = []
        
        for char_profile in character_profiles:
            basic_info = char_profile.get('basic_info', {})
            
            # 获取人物名称
            char_name = char_profile.get('name', '') or basic_info.get('name', '')
            if not char_name:
                continue
            
            # 检查人物是否出现（支持姓名变体）
            name_variants = self._get_character_name_variants(char_profile, basic_info)
            found_in_text = any(variant in text for variant in name_variants if variant)
            if not found_in_text:
                continue
            
            # === 基础分：人物存在 = 0.5分 ===
            base_score = 0.5
            
            # === 1. 提取性格关键词（智能提取）===
            personality_keywords = []
            
            # 从personality字段提取（支持markdown格式）
            personality = char_profile.get('personality', '') or basic_info.get('personality', '')
            if personality:
                # 清理markdown标记
                clean_personality = re.sub(r'\*\*[^*]+\*\*[:：]', '', personality)
                clean_personality = re.sub(r'\*\*', '', clean_personality)
                
                # 提取关键词
                keywords = re.split(r'[,，。、；;\\n]+\s*', clean_personality)
                for kw in keywords:
                    kw = kw.strip()
                    if kw and 2 <= len(kw) <= 10:
                        personality_keywords.append(kw)
            
            # 从traits字段提取
            traits = char_profile.get('traits', []) or basic_info.get('traits', [])
            if traits:
                if isinstance(traits, list):
                    personality_keywords.extend([t for t in traits if t and 2 <= len(t) <= 10])
                elif isinstance(traits, str):
                    keywords = re.split(r'[,，。、；;]+\s*', traits)
                    personality_keywords.extend([k.strip() for k in keywords if k.strip() and 2 <= len(k.strip()) <= 10])
            
            # 去重
            personality_keywords = list(set([k for k in personality_keywords if 2 <= len(k) <= 10]))
            
            # === 2. 获取人物相关段落 ===
            char_segments = []
            for variant in name_variants:
                if variant and variant in text:
                    pos = 0
                    while True:
                        pos = text.find(variant, pos)
                        if pos == -1:
                            break
                        start = max(0, pos - 300)
                        end = min(len(text), pos + len(variant) + 300)
                        char_segments.append(text[start:end])
                        pos += 1
                        if len(char_segments) >= 5:
                            break
                    if len(char_segments) >= 5:
                        break
            
            # === 3. 计算性格关键词匹配 ===
            keyword_score = 0.0
            matched_keywords = []
            unmatched_keywords = []
            
            if personality_keywords and char_segments:
                char_text = '\n'.join(char_segments)
                
                for keyword in personality_keywords:
                    # 直接匹配
                    if keyword in char_text:
                        matched_keywords.append(keyword)
                    else:
                        # 语义扩展匹配
                        semantic_matches = self._get_semantic_matches(keyword)
                        if any(sm in char_text for sm in semantic_matches):
                            matched_keywords.append(keyword)
                        else:
                            unmatched_keywords.append(keyword)
                
                if personality_keywords:
                    match_ratio = len(matched_keywords) / len(personality_keywords)
                    keyword_score = match_ratio * 0.3
            elif not personality_keywords:
                keyword_score = 0.15
            
            # === 4. 计算行为特征匹配 ===
            behavior_score = 0.0
            dialogue_score = 0.0
            
            if char_segments:
                char_text = '\n'.join(char_segments)
                
                # 检查人物对话
                has_dialogue = any(marker in char_text for marker in ['"', '"', '「', '」', '『', '』'])
                if has_dialogue:
                    dialogue_score = 0.1
                
                # 检查人物行为描写
                action_patterns = ['说', '做', '走', '看', '想', '拿', '放', '转身', '抬头', '低头', '走进', '走出']
                action_count = sum(1 for p in action_patterns if p in char_text)
                if action_count >= 3:
                    behavior_score = 0.1
            
            # === 5. 人物提及频率加分 ===
            mention_score = 0.0
            total_mentions = 0
            for variant in name_variants:
                if variant:
                    total_mentions += text.count(variant)
            
            if total_mentions >= 5:
                mention_score = 0.1
            elif total_mentions >= 3:
                mention_score = 0.05
            
            # === 6. 计算总分 ===
            total_score = min(1.0, base_score + keyword_score + behavior_score + dialogue_score + mention_score)
            
            # 确保最低分不低于0.5
            total_score = max(0.5, total_score)
            
            consistency_scores.append(total_score)
            
            # 记录问题
            if total_score < 0.7 and unmatched_keywords:
                missing = unmatched_keywords[:3]
                character_issues.append(f"{char_name}的性格特征'{', '.join(missing)}'未充分体现")
        
        if not consistency_scores:
            return 0.7
        
        final_score = sum(consistency_scores) / len(consistency_scores)
        
        if character_issues:
            self.logger.info(f"[V5.5] 人设问题: {'; '.join(character_issues[:3])}")
        
        return final_score
    
    def _get_semantic_matches(self, keyword: str) -> List[str]:
        """获取性格关键词的语义匹配词（近义词、相关词）"""
        semantic_map = {
            '乐观': ['笑着', '开心', '高兴', '快乐', '笑容', '开朗', '积极'],
            '开朗': ['笑着', '开心', '热情', '活泼', '笑容'],
            '坚韧': ['坚持', '不放弃', '咬牙', '硬撑', '毅力'],
            '善良': ['帮', '关心', '照顾', '心疼', '温柔'],

    def _score_worldview_consistency(self, text: str, world_view: str) -> Tuple[float, bool]:
        """世界观一致性评分（权重：20%，严重违背时一票否决）"""
        if not world_view:
            return 0.7, False
        
        # 提取世界观关键设定
        worldview_elements = self._extract_worldview_elements(world_view)
        
        # 检查是否严重违背
        for element_type, elements in worldview_elements.items():
            if element_type == '题材类型':
                for element in elements:
                    if '现实' in element:
                        # 检查是否出现魔法、超自然元素
                        fantasy_keywords = ['魔法', '法术', '异能', '修仙', '仙术', '神通']
                        if any(kw in text for kw in fantasy_keywords):
                            self.logger.warning("检测到现实题材中出现魔法元素，严重违背世界观")
                            return 0.0, True  # 一票否决
        
        # 一般违背检测
        violation_count = 0
        total_elements = 0
        
        for element_type, elements in worldview_elements.items():
            total_elements += len(elements)
            for element in elements:
                element_keywords = self._extract_keywords(element)
                if any(kw in text for kw in element_keywords):
                    violation_count -= 1
        
        # 计算分数
        if total_elements == 0:
            return 0.7, False
        
        violation_ratio = max(0, violation_count) / total_elements
        
        if violation_ratio == 0:
            return 1.0, False
        elif violation_ratio <= 0.3:
            return 0.5, False
        else:
            return 0.0, True
    
    def _score_naturalness(self, text: str) -> NaturalnessScore:
        """自然度评分（权重：5%）"""
        # AI痕迹检测
        detected_patterns = []
        pattern_scores = {}
        
        for category, patterns in self.ai_patterns.items():
            category_detections = []
            
            for pattern in patterns:
                try:
                    matches = re.findall(pattern, text)
                    if matches:
                        category_detections.append(f"{category}: {pattern}")
                except re.error:
                    continue
            
            if category_detections:
                detected_patterns.extend(category_detections)
                pattern_scores[category] = len(category_detections)

        # 陈词滥调检测
        import jieba
        cliche_count = 0
        words = list(jieba.cut(text))
        for cliche in self.cliches:
            cliche_count += words.count(cliche)
        
        total_words = len(words)
        cliche_score = cliche_count / total_words if total_words > 0 else 0.0
        
        # 公式化程度
        formulaic_score = self._calculate_formulaic_score(text)
        
        # AI概率
        total_patterns = sum(pattern_scores.values())
        pattern_probability = min(total_patterns / 10, 1.0)
        ai_probability = (
            pattern_probability * 0.4 +
            formulaic_score * 0.4 +
            cliche_score * 0.2
        )
        
        # 评分
        if ai_probability <= 0.2:
            score = 1.0
        elif ai_probability <= 0.4:
            score = 0.8
        elif ai_probability <= 0.6:
            score = 0.6
        else:
            score = 0.4
        
        # 生成问题列表
        issues = []
        if ai_probability > 0.4:
            issues.append(f"AI生成概率较高: {ai_probability:.1%}")
        if formulaic_score > 0.6:
            issues.append("公式化程度较高")
        if cliche_score > 0.05:
            issues.append("陈词滥调较多")
        
        return NaturalnessScore(
            score=score,
            ai_probability=ai_probability,
            formulaic_score=formulaic_score,
            cliche_score=cliche_score,
            issues_found=issues
        )

    # 辅助方法
    def _count_words(self, text: str) -> int:
        """统计字数"""
        chinese_chars = len(re.findall(r'[\u4e00-\u9fff]', text))
        english_chars = len(re.findall(r'[a-zA-Z]', text))
        english_words = english_chars // 5
        digit_chars = len(re.findall(r'\d', text))
        digit_words = digit_chars // 10
        return chinese_chars + english_words + digit_words
    
    def _count_words_excluding_marker(self, text: str) -> int:
        """统计字数（排除【本章完】标记）"""
        clean_text = text.replace("【本章完】", "")
        return self._count_words(clean_text)
    
    def _extract_plot_points(self, chapter_outline: str) -> List[str]:
        """从大纲中提取关键情节点"""
        if not chapter_outline:
            return []
        
        key_points = []
        lines = chapter_outline.split('\n')
        
        for line in lines:
            line = line.strip()
            
            if line.startswith('#### 第') or line.startswith('## 第') or line.startswith('# 第'):
                continue
            
            if not line:
                continue
            
            if line.startswith('**') and line.endswith('**') and len(line) < 20:
                continue
            
            if len(line) < 4:
                continue
            
            if any(c in line for c in ['后', '时', '了', '的', '是', '在', '去', '来', '到', '从', '向']):
                key_points.append(line)
        
        if not key_points:
            sentences = re.split(r'[。！？\n]', chapter_outline)
            for sentence in sentences:
                sentence = sentence.strip()
                if len(sentence) >= 4 and any(c in sentence for c in ['第', '章', '后', '时', '了', '的']):
                    key_points.append(sentence)
        
        return key_points[:10]
    
    def _extract_keywords(self, text: str) -> List[str]:
        """提取关键词"""
        import jieba
        words = list(jieba.cut(text))
        keywords = []

        stopwords = {'的', '了', '在', '是', '我', '有', '和', '就', '不', '人', '都', '一',
                     '一个', '上', '也', '很', '到', '说', '要', '去', '你', '会', '着', '没有',
                     '看', '好', '自己', '这', '那'}
        
        for word in words:
            if len(word) >= 2 and word not in stopwords and word.isalpha():
                keywords.append(word)
        
        word_freq = Counter(keywords)
        return [word for word, _ in word_freq.most_common(20)]
    
    def _extract_worldview_elements(self, world_view: str) -> Dict[str, List[str]]:
        """提取世界观元素"""
        elements = {}
        
        sections = world_view.split('\n\n')
        
        for section in sections:
            if not section.strip():
                continue
            
            lines = section.split('\n')
            if not lines:
                continue
            
            title = lines[0].strip()
            content = '\n'.join(lines[1:])
            
            if title not in elements:
                elements[title] = []
            
            if content:
                elements[title].append(content)
        
        return elements
    
    def _calculate_formulaic_score(self, text: str) -> float:
        """计算公式化程度"""
        formulaic_patterns = [
            r'在.*的.*下',
            r'从.*的.*角度',
            r'通过.*的.*方式',
            r'以.*为.*的.*',
            r'将.*与.*相结合'
        ]
        
        formulaic_count = 0
        for pattern in formulaic_patterns:
            formulaic_count += len(re.findall(pattern, text))
        
        sentences = self._split_sentences(text)
        if sentences:
            score = min(formulaic_count / len(sentences), 1.0)
        else:
            score = 0.0
        
        return score
    
    def _split_sentences(self, text: str) -> List[str]:
        """分割句子"""
        sentences = []
        current_sentence = []
        
        for char in text:
            current_sentence.append(char)
            
            if char in '。！？.!?':
                sentence = ''.join(current_sentence).strip()
                if sentence:
                    sentences.append(sentence)
                current_sentence = []
        
        if current_sentence:
            sentence = ''.join(current_sentence).strip()
            if sentence:
                sentences.append(sentence)
        
        return sentences
    
    def _get_character_name_variants(self, char_profile: Dict, basic_info: Dict) -> List[str]:
        """获取人物姓名变体（昵称、别名等）"""
        variants = []
        
        main_name = char_profile.get('name', '') or basic_info.get('name', '')
        if main_name:
            variants.append(main_name)
        
        nickname = basic_info.get('nickname', '') or char_profile.get('nickname', '')
        if nickname:
            variants.append(nickname)
        
        if main_name and len(main_name) > 1:
            if '\u4e00' <= main_name[-1] <= '\u9fff':
                variants.append(main_name[-2:] if len(main_name) >= 2 else main_name[-1])
        
        return variants
    
    def _extract_personality_keywords(self, text: str) -> List[str]:
```

## 文件二：iterative_generator_v2.py

**路径**: `scripts/iterative_generator_v2.py`

```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
迭代生成器 V2 - 完全按照用户要求重写

核心流程（点下【开始生成】后的运行逻辑）：
1. 整理打包请求内容（百分百附加要求：结束加上【本章完】）
2. 向大模型发送请求（本地调用模型/线上调用API）
3. 接受返回文章
4. 从多维度评分（世界观、大纲、风格、人设、Ai感、字数、上下文契合度，没有【本章完】视作未完成）
5. 分数小于0.8 -> 发送评分（总评分+各维度评分）+ 修改建议（围绕各维度） -> 再次接受返还
6. 再次评分 -> ... -> 评分大于0.8且标记【本章完】
7. 输出保存
"""

import sys
import logging
from typing import Dict, List, Optional, Tuple
from dataclasses import dataclass
from enum import Enum
import re


class GenerationStrategy(Enum):
    """生成策略"""
    CREATIVE = "creative"      # 创意优先（高温度）
    BALANCED = "balanced"      # 平衡（中等温度）
    PRECISE = "precise"        # 精确优先（低温度）


@dataclass
class DimensionScore:
    """单维度评分"""
    dimension_name: str      # 维度名称
    score: float             # 评分 0.0-1.0
    details: str             # 详细说明
    issues: List[str]        # 发现的问题


@dataclass
class IterationResult:
    """迭代结果"""
    iteration: int           # 迭代轮次
    content: str             # 生成内容
    total_score: float      # 总评分
    dimension_scores: Dict[str, DimensionScore]  # 各维度评分
    has_chapter_end: bool    # 是否有【本章完】标记
    feedback: str            # 反馈内容
    suggestions: List[str]  # 修改建议


class IterativeGeneratorV2:
    """
    迭代生成器 V2 - 完全重写版本

    严格按照用户要求的流程实现：
    1. 打包请求内容（必须包含【本章完】要求）
    2. 发送请求生成
    3. 多维度评分
    4. 评分<0.8或缺少【本章完】 -> 循环优化
    5. 评分>=0.8且有【本章完】 -> 输出保存
    """

    def __init__(
        self,
        api_client,
        logger: logging.Logger,
        model_name: str = "deepseek-chat",
        target_word_count: int = 3500,
        quality_threshold: float = 0.8,
        max_iterations: int = 5
    ):
        """
        初始化迭代生成器 V2

        Args:
            api_client: API客户端
            logger: 日志记录器
            model_name: 模型名称
            target_word_count: 目标字数
            quality_threshold: 质量阈值（默认0.8）
            max_iterations: 最大迭代次数
        """
        self.api_client = api_client
        self.logger = logger
        self.model_name = model_name
        self.target_word_count = target_word_count
        self.quality_threshold = quality_threshold
        self.max_iterations = max_iterations

        # 生成历史（用于调试和追踪）
        self.generation_history: List[IterationResult] = []

        self.logger.info(f"[V2] 迭代生成器初始化完成")
        self.logger.info(f"[V2] 目标字数: {target_word_count}, 质量阈值: {quality_threshold}, 最大迭代: {max_iterations}")

    def generate_with_iteration(
        self,
        prompt: str,
        validation_fn,
        strategy: GenerationStrategy = GenerationStrategy.BALANCED
    ) -> Tuple[str, Dict]:
        """
        完整的迭代生成流程

        流程：
        1. 整理打包请求内容（强制附加【本章完】要求）
        2. 发送请求生成内容
        3. 多维度评分
        4. 如果评分<0.8或缺少【本章完】标记：
           - 构建反馈（评分+各维度评分+修改建议）
           - 再次生成
        5. 循环直到评分>=0.8且有【本章完】标记
        6. 输出保存

        Args:
            prompt: 基础提示词
            validation_fn: 验证函数，返回 (total_score, dimension_scores, suggestions)
            strategy: 生成策略

        Returns:
            (最终生成内容, 统计信息)
        """
        print(f"[V2][INFO] ========== 开始迭代生成流程 ==========", file=sys.stderr)
        print(f"[V2][INFO] 质量阈值: {self.quality_threshold}, 最大迭代: {self.max_iterations}", file=sys.stderr)
        self.logger.info("[V2] ========== 开始迭代生成流程 ==========")

        best_result = None
        best_score = 0.0

        stats = {
            'iterations': 0,
            'scores': [],
            'dimension_scores': [],
            'has_chapter_end_history': [],
            'feedback_history': []
        }

        # 开始迭代
        for iteration in range(self.max_iterations):
            print(f"[V2][ITERATION] ========== 第 {iteration + 1} 轮迭代 ==========", file=sys.stderr)
            self.logger.info(f"[V2] 第 {iteration + 1} 轮迭代开始")

            # === 步骤1: 整理打包请求内容 ===
            current_prompt = self._build_request_prompt(prompt, iteration, best_result)

            # === 步骤2: 向大模型发送请求 ===
            print(f"[V2][INFO] 正在向大模型发送请求...", file=sys.stderr)
            self.logger.info(f"[V2] 正在发送API请求...")

            try:
                generated_content = self._send_request_to_model(current_prompt, strategy)
            except Exception as e:
                self.logger.error(f"[V2] API请求失败: {e}")
                if iteration == 0:
                    raise
                print(f"[V2][WARNING] 第{iteration + 1}轮生成失败，使用上一轮结果", file=sys.stderr)
                break

            # === 步骤3: 接受返回文章 ===
            print(f"[V2][INFO] 接收到返回内容，长度: {len(generated_content)} 字符", file=sys.stderr)
            self.logger.info(f"[V2] 接收到返回内容，长度: {len(generated_content)} 字符")

            # === 步骤4: 从多维度评分 ===
            print(f"[V2][INFO] 开始多维度评分...", file=sys.stderr)
            total_score, dimension_scores, suggestions = self._evaluate_content(
                generated_content, validation_fn
            )

            # 检查【本章完】标记
            has_chapter_end = '【本章完】' in generated_content
            print(f"[V2][INFO] 【本章完】检查: {'✓ 有' if has_chapter_end else '✗ 没有'}", file=sys.stderr)
            print(f"[V2][INFO] 总评分: {total_score:.3f}", file=sys.stderr)

            # 打印各维度评分
            for dim_name, dim_score in dimension_scores.items():
                print(f"[V2][INFO]   {dim_name}: {dim_score.score:.3f} - {dim_score.details}", file=sys.stderr)

            # 记录结果
            iteration_result = IterationResult(
                iteration=iteration + 1,
                content=generated_content,
                total_score=total_score,
                dimension_scores=dimension_scores,
                has_chapter_end=has_chapter_end,
                feedback=self._build_feedback_text(total_score, dimension_scores, has_chapter_end),
                suggestions=suggestions
            )

            self.generation_history.append(iteration_result)
            stats['iterations'] += 1
            stats['scores'].append(total_score)
            stats['dimension_scores'].append(dimension_scores)
            stats['has_chapter_end_history'].append(has_chapter_end)
            stats['feedback_history'].append(iteration_result.feedback)

            # === 步骤5: 判断是否满足停止条件 ===
            # 检查字数偏差（超过50%强制继续迭代）
            word_count_score = dimension_scores.get('字数')
            word_count_deviation = 0
            if word_count_score and isinstance(word_count_score, DimensionScore):
                word_count_deviation = 1 - word_count_score.score
            
            # 必须同时满足：评分 >= 0.8、有【本章完】标记、字数偏差不超过50%
            word_count_acceptable = word_count_score is None or (isinstance(word_count_score, DimensionScore) and word_count_score.score >= 0.3)
            
            if total_score >= self.quality_threshold and has_chapter_end and word_count_acceptable:
                print(f"[V2][SUCCESS] ========== 迭代完成！满足条件 ==========", file=sys.stderr)
                print(f"[V2][SUCCESS] 总评分: {total_score:.3f} >= {self.quality_threshold}", file=sys.stderr)
                print(f"[V2][SUCCESS] 包含【本章完】标记", file=sys.stderr)
                self.logger.info(f"[V2] 迭代完成！满足条件，评分: {total_score:.3f}")

                best_result = iteration_result
                best_score = total_score
                break

            # === 不满足条件，构建反馈，准备下一轮 ===
            print(f"[V2][CONTINUE] 未满足条件，准备第 {iteration + 2} 轮迭代", file=sys.stderr)
            print(f"[V2][CONTINUE] 原因: ", file=sys.stderr)
            if total_score < self.quality_threshold:
                print(f"[V2][CONTINUE]   - 评分 {total_score:.3f} < 阈值 {self.quality_threshold}", file=sys.stderr)
            if not has_chapter_end:
                print(f"[V2][CONTINUE]   - 缺少【本章完】标记", file=sys.stderr)
            if not word_count_acceptable:
                print(f"[V2][CONTINUE]   - 字数偏差过大（评分 {word_count_score.score:.3f} < 0.3）", file=sys.stderr)

            # 打印反馈内容
            print(f"[V2][FEEDBACK] 反馈内容:", file=sys.stderr)
            print(iteration_result.feedback, file=sys.stderr)

            # 如果是最佳结果，保存
            if total_score > best_score:
                best_result = iteration_result
                best_score = total_score

        # 迭代结束
        print(f"[V2][FINAL] ========== 迭代结束 ==========", file=sys.stderr)
        print(f"[V2][FINAL] 总迭代次数: {stats['iterations']}", file=sys.stderr)
        print(f"[V2][FINAL] 最佳评分: {best_score:.3f}", file=sys.stderr)

        # 输出保存
        final_content = best_result.content if best_result else ""
        print(f"[V2][SAVE] 最终内容长度: {len(final_content)} 字符", file=sys.stderr)
        print(f"[V2][SAVE] 最终内容包含【本章完】: {'是' if '【本章完】' in final_content else '否'}", file=sys.stderr)

        self.logger.info(f"[V2] 迭代结束，返回最终内容")
        return final_content, stats

    def _build_request_prompt(
        self,
        base_prompt: str,
        iteration: int,
        previous_result: Optional[IterationResult]
    ) -> str:
        """
        步骤1: 整理打包请求内容

        强制要求：
        - 结尾必须加上【本章完】
        - 如果不是第一轮，包含上一轮的反馈和改进建议
        """
        # 第一轮：基础提示词 + 强制要求
        if iteration == 0:
            prompt = base_prompt.strip()

            # 强制附加【本章完】要求（如果没有的话）
            if "【本章完】" not in prompt:
                prompt += "\n\n重要要求：章节结束时必须在末尾添加【本章完】标记！"

            print(f"[V2][PROMPT] 第一轮提示词（前200字符）: {prompt[:200]}...", file=sys.stderr)
            return prompt

        # 后续轮次：基础提示词 + 反馈 + 改进要求
        print(f"[V2][PROMPT] 构建第{iteration + 1}轮提示词...", file=sys.stderr)

        prompt_parts = [
            base_prompt.strip(),
            "",
            "=" * 60,
            "【上一轮反馈与改进要求】",
            ""
        ]

        # 添加评分信息
        if previous_result:
            prompt_parts.append(f"上一轮总评分: {previous_result.total_score:.3f} / 1.0")
            prompt_parts.append("各维度评分:")
            for dim_name, dim_score in previous_result.dimension_scores.items():
                prompt_parts.append(f"  - {dim_name}: {dim_score.score:.3f} - {dim_score.details}")

            prompt_parts.append("")
            prompt_parts.append("【详细反馈】")
            prompt_parts.append(previous_result.feedback)

            prompt_parts.append("")
            prompt_parts.append("【改进建议】")
            for idx, suggestion in enumerate(previous_result.suggestions, 1):
                prompt_parts.append(f"{idx}. {suggestion}")

        # 强制要求【本章完】和字数
        prompt_parts.extend([
            "",
            "=" * 60,
            "特别强调:",
            f"1. 必须按照上述反馈进行改进",
            f"2. 【目标字数】{self.target_word_count}字，字数必须接近目标（误差±10%以内）",
            f"3. 章节结束时**必须**在末尾添加【本章完】标记！",
            f"4. 这是停止生成的必要条件",
            ""
        ])

        final_prompt = "\n".join(prompt_parts)
        print(f"[V2][PROMPT] 提示词长度: {len(final_prompt)} 字符", file=sys.stderr)

        return final_prompt

    def _send_request_to_model(
        self,
        prompt: str,
        strategy: GenerationStrategy
    ) -> str:
        """
        步骤2: 向大模型发送请求

        支持本地模型和在线API
        """
        # 计算max_tokens - V5.7修复：确保AI有足够token生成完整章节
        base_tokens = int(self.target_word_count / 1.5)
        max_tokens = int(base_tokens * 1.5)  # 允许50%浮动
        max_tokens = max(max_tokens, 1000)
        max_tokens = min(max_tokens, 4096)

        # 根据策略设置温度
        temperature_map = {
            GenerationStrategy.CREATIVE: 0.9,
            GenerationStrategy.BALANCED: 0.7,
            GenerationStrategy.PRECISE: 0.5
        }
        temperature = temperature_map.get(strategy, 0.7)

        print(f"[V2][API] 发送请求 - max_tokens: {max_tokens}, temperature: {temperature}", file=sys.stderr)

        # 构建强化的system prompt
        system_prompt = """你是一位经验丰富的小说创作专家,必须严格遵守以下核心原则:

【核心原则】
1. **人物设定不可违背**: 
   - 人物的性格、外貌、背景、行为模式必须严格遵循设定
   - 人物的对话风格、语言习惯必须与设定一致
   - 人物之间的关系必须符合设定
   - 严禁擅自改变人物设定或添加不符合设定的行为

2. **大纲严格执行**: 
   - 必须完整执行章节大纲的所有要点
   - 不得遗漏关键情节

3. **风格保持一致**: 
   - 在遵守设定的前提下,保持与提供风格样本相同的叙事风格

【人物设定优先级】
- 人物设定 > 风格模仿
- 设定准确性 > 文学修辞
- 人物一致性 > 情节戏剧性

【输出要求】
- 必须以【本章完】结尾
- 字数必须接近目标字数(误差±10%以内)
- 人物言行必须100%符合设定"""

        # 调用API客户端
        try:
            response = self.api_client.chat.completions.create(
                model=self.model_name,
                messages=[
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": prompt}
                ],
                temperature=temperature,
                max_tokens=max_tokens,
                timeout=120
            )

            content = response.choices[0].message.content

            print(f"[V2][API] API响应成功，内容长度: {len(content)} 字符", file=sys.stderr)

            return content

        except Exception as e:
            print(f"[V2][ERROR] API调用失败: {e}", file=sys.stderr)
            self.logger.error(f"[V2] API调用失败: {e}")
            raise

    def _evaluate_content(
        self,
        content: str,
        validation_fn
    ) -> Tuple[float, Dict[str, DimensionScore], List[str]]:
        """
        步骤4: 从多维度评分
        """
        try:
            total_score, detailed_scores, suggestions = validation_fn(content)

            if not isinstance(total_score, (int, float)):
                total_score = float(total_score)

            if not isinstance(detailed_scores, dict):
                detailed_scores = {}

            if not isinstance(suggestions, list):
                suggestions = list(suggestions) if suggestions else []

            # 创建DimensionScore对象
            dimension_scores = {}
            for dim_name, dim_data in detailed_scores.items():
                if isinstance(dim_data, DimensionScore):
                    dimension_scores[dim_name] = dim_data
                elif isinstance(dim_data, dict):
                    dimension_scores[dim_name] = DimensionScore(
                        dimension_name=dim_name,
                        score=float(dim_data.get('score', 0.5)),
                        details=dim_data.get('details', ''),
                        issues=dim_data.get('issues', [])
                    )
                elif isinstance(dim_data, (int, float)):
                    dimension_scores[dim_name] = DimensionScore(
                        dimension_name=dim_name,
                        score=float(dim_data),
                        details='',
                        issues=[]
                    )
                else:
                    dimension_scores[dim_name] = DimensionScore(
                        dimension_name=dim_name,
                        score=0.5,
                        details=f'数据类型错误: {type(dim_data)}',
                        issues=[]
                    )

            print(f"[V2][SCORE] 评分完成 - 总分: {total_score:.3f}, 维度数: {len(dimension_scores)}", file=sys.stderr)

            return total_score, dimension_scores, suggestions

        except Exception as e:
            print(f"[V2][ERROR] 评分过程出错: {e}", file=sys.stderr)
            self.logger.error(f"[V2] 评分过程出错: {e}")

            default_scores = {
                '世界观': DimensionScore('世界观', 0.5, '评分失败', []),
                '大纲': DimensionScore('大纲', 0.5, '评分失败', []),
                '风格': DimensionScore('风格', 0.5, '评分失败', []),
                '人设': DimensionScore('人设', 0.5, '评分失败', []),
                'AI感': DimensionScore('AI感', 0.5, '评分失败', []),
                '字数': DimensionScore('字数', 0.5, '评分失败', []),
                '上下文契合度': DimensionScore('上下文契合度', 0.5, '评分失败', [])
            }

            return 0.5, default_scores, ['评分过程出错，请检查日志']

    def _build_feedback_text(
        self,
        total_score: float,
        dimension_scores: Dict[str, DimensionScore],
        has_chapter_end: bool
    ) -> str:
        """
        步骤5: 构建反馈文本（增强版 - 针对性改进建议）

        格式：总评分 + 各维度评分 + 问题描述(优先显示设定偏离问题) + 具体改进建议
        """
        feedback_parts = [
            f"【总评分】: {total_score:.3f} / 1.0",
            ""
        ]

        # 优先级1:检查设定偏离(人设、世界观、风格)
        priority_dims = ['人设', '世界观', '风格']
        setting_issues = []

        for dim_name in priority_dims:
            if dim_name in dimension_scores:
                dim_score = dimension_scores[dim_name]
                if dim_score.score < 0.7:
                    setting_issues.append((dim_name, dim_score))

        if setting_issues:
            feedback_parts.append("=" * 60)
            feedback_parts.append("🔴 【核心问题 - 设定偏离】(必须立即修正)")
            feedback_parts.append("=" * 60)

            for dim_name, dim_score in setting_issues:
                feedback_parts.append(f"⚠️ {dim_name}不符合: 评分 {dim_score.score:.3f} (要求≥0.7)")
                if dim_score.details:
                    feedback_parts.append(f"   说明: {dim_score.details}")
                if dim_score.issues:
                    feedback_parts.append(f"   问题:")
                    for issue in dim_score.issues:
                        feedback_parts.append(f"     - {issue}")

            feedback_parts.append("")
            feedback_parts.append("【修正要求】")
            feedback_parts.append("  ❌ 严禁偏离提供的人物设定、世界观、写作风格")
            feedback_parts.append("  ✅ 必须严格按照原始设定重新创作相关内容")
            feedback_parts.append("")
```

```python
    def _ensure_chapter_end_marker(self, prompt: str) -> str:
        """
        确保提示词中包含【本章完】要求

        这是百分百必须的附加要求
        """
        if "【本章完】" in prompt:
            if "必须" not in prompt or "强制" not in prompt:
                prompt += "\n\n【重要提醒】章节结束时必须在末尾添加【本章完】标记！"
        else:
            prompt += "\n\n【重要要求】章节结束时必须在末尾添加【本章完】标记！\n这是章节完成的必要条件，请务必遵守。"

        return prompt
    
    def _score_context_fit(self, content: str, previous_chapters: List[str]) -> float:
        """
        上下文契合度评分
        
        分析当前章节与前面章节的一致性
        """
        if not previous_chapters:
            return 0.8
        
        scores = []
        import re
        
        previous_text = '\n'.join(previous_chapters)
        
        # 1. 人物名称一致性
        prev_names = set(re.findall(r'[\u4e00-\u9fa5]{2,4}', previous_text))
        curr_names = set(re.findall(r'[\u4e00-\u9fa5]{2,4}', content))
        
        stop_words = {'世界', '中国', '大陆', '森林', '山脉', '河流', '宫殿', '国家', '军队', '人民', '百姓', '江湖', '武林', '修仙'}
        prev_names = prev_names - stop_words
        curr_names = curr_names - stop_words
        
        if prev_names:
            name_overlap = len(prev_names & curr_names) / len(prev_names)
            scores.append(min(1.0, name_overlap + 0.2))
        else:
            scores.append(0.8)
        
        # 2. 关键事件延续性
        action_patterns = ['说', '走', '看', '想', '做', '去', '来', '到', '发现', '决定', '明白', '意识到']
        prev_actions = set(a for a in action_patterns if a in previous_text)
        curr_actions = set(a for a in action_patterns if a in content)
        
        if prev_actions:
            action_overlap = len(prev_actions & curr_actions) / len(prev_actions)
            scores.append(min(1.0, action_overlap + 0.2))
        else:
            scores.append(0.8)
        
        # 3. 时间线连贯性
        time_words = ['之后', '然后', '接着', '后来', '第二天', '今天', '昨天', '明天']
        prev_time = sum(1 for w in time_words if w in previous_text)
        curr_time = sum(1 for w in time_words if w in content)
        
        if prev_time > 0 and curr_time > 0:
            scores.append(0.9)
        else:
            scores.append(0.7)
        
        # 加权平均
        weights = [0.4, 0.35, 0.25]
        final_score = sum(s * w for s, w in zip(scores, weights))
        
        return final_score
```

```python
智能上下文构建器 - 最小化Token使用

import re
import logging
from typing import Dict, List, Optional, Set, Tuple
from collections import defaultdict


class SmartContextBuilder:
    """智能上下文构建器 - 最小化Token使用"""

    def __init__(self, logger: logging.Logger):
        """初始化上下文构建器"""
        self.logger = logger
        self.context_memory = {}  # 上下文记忆，避免重复发送
        self.entity_cache = {}    # 实体缓存

    def build_optimized_prompt(
        self,
        chapter_title: str,
        chapter_outline: str,
        world_view: str,
        style: str,
        characters: List[Dict],
        previous_chapters: Optional[List[str]] = None,
        max_worldview_tokens: int = 2000,
        max_style_tokens: int = 1500,
        target_word_count: int = 3500
    ) -> str:
        """
        构建优化的提示词（基于七要素提示词工程框架）

        改进点：
        1. 明确角色与语气
        2. 增强输出要求的可见性（使用emoji和分隔线）
        3. 添加叙事目标
        4. 提供创作检查清单
        5. 优化冲突钩子提取
        """
        prompt_parts = []

        # ========== 第零部分：核心创作原则(最高优先级) ==========
        prompt_parts.append("=" * 60)
        prompt_parts.append("🔴 核心创作原则(最高优先级,违反将导致创作失败)")
        prompt_parts.append("=" * 60)
        prompt_parts.append("在开始创作前,你必须:")
        prompt_parts.append("  1. **严格**按照提供的人物设定塑造角色")
        prompt_parts.append("  2. **严格**遵守世界观的所有设定")
        prompt_parts.append("  3. **严格**遵循写作风格指导")
        prompt_parts.append("  4. **完整**执行章节大纲的所有要点")
        prompt_parts.append("")
        prompt_parts.append("【重要警告】")
        prompt_parts.append("⚠️ 擅自修改设定将导致评分不及格,需要重新生成")
        prompt_parts.append("⚠️ 偏离人物设定将导致评分不及格")
        prompt_parts.append("⚠️ 违背世界观将导致评分不及格")
        prompt_parts.append("")

        # ========== 第一部分：角色与语气 ==========
        prompt_parts.append("=" * 60)
        prompt_parts.append("创作任务")
        prompt_parts.append("=" * 60)
        prompt_parts.append(f"请创作小说章节：{chapter_title}")
        prompt_parts.append("")

        # ========== 第二部分：核心输出要求（最突出的部分）==========
        prompt_parts.append("=" * 60)
        prompt_parts.append("⚠️ 核心输出要求（违反将导致生成失败）⚠️")
        prompt_parts.append("=" * 60)
        prompt_parts.append(f"1. 【最重要】章节内容**必须**以【本章完】结尾，这是章节完成的唯一标志！")
        tolerance_low = int(target_word_count * 0.9)
        tolerance_high = int(target_word_count * 1.1)
        prompt_parts.append(f"2. 【字数硬性要求】目标字数：{target_word_count}字")
        prompt_parts.append(f"   - 合理范围：{tolerance_low}-{tolerance_high}字（±10%）")
        prompt_parts.append(f"   - ⚠️ 超过{tolerance_high}字将触发扣分，超标50%以上直接判定失败！")
        prompt_parts.append(f"3. 【一票否决】如果没有【本章完】标记，评分将为0，需要重新生成！")
        prompt_parts.append("")

        # ========== 第三部分：章节大纲 ==========
        if chapter_outline:
            prompt_parts.append("=" * 60)
            prompt_parts.append("章节大纲")
            prompt_parts.append("=" * 60)
            prompt_parts.append(chapter_outline)

            # 提取冲突钩子
            conflict_hooks = self._extract_conflict_hooks(chapter_outline)
            if conflict_hooks:
                prompt_parts.append("")
                prompt_parts.append("🎯 核心冲突与张力：")
                for hook in conflict_hooks:
                    prompt_parts.append(f"  • {hook}")
            prompt_parts.append("")

        # ========== 第四部分：人物设定 ==========
        relevant_characters = self._retrieve_relevant_characters(
            chapter_outline, characters
        )
        if relevant_characters:
            prompt_parts.append("=" * 60)
            prompt_parts.append("人物设定")
            prompt_parts.append("=" * 60)
            prompt_parts.append("（本章涉及的主要人物）")
            for char_desc in relevant_characters:
                prompt_parts.append(f"\n{char_desc}")
            prompt_parts.append("")

        # ========== 第五部分：世界观背景 ==========
        if world_view:
            relevant_worldview = self._retrieve_relevant_worldview(
                chapter_outline, world_view, max_worldview_tokens
            )
            if relevant_worldview:
                prompt_parts.append("=" * 60)
                prompt_parts.append("世界观背景")
                prompt_parts.append("=" * 60)
                prompt_parts.append(relevant_worldview)
                prompt_parts.append("")

        # ========== 第六部分：写作风格 ==========
        if style:
            compressed_style = self._compress_style(style, max_style_tokens)
            if compressed_style:
                prompt_parts.append("=" * 60)
                prompt_parts.append("写作风格")
                prompt_parts.append("=" * 60)
                prompt_parts.append(compressed_style)
                prompt_parts.append("")

        # ========== 第七部分：前情提要 ==========
        if previous_chapters:
            context_summary = self._build_context_summary(previous_chapters)
            if context_summary:
                prompt_parts.append("=" * 60)
                prompt_parts.append("前情提要")
                prompt_parts.append("=" * 60)
                prompt_parts.append(context_summary)
                prompt_parts.append("")

        # ========== 第八部分：创作检查清单 ==========
        prompt_parts.append("=" * 60)
        prompt_parts.append("创作检查清单")
        prompt_parts.append("=" * 60)
        prompt_parts.append("在创作前请确认：")
        prompt_parts.append(f"  [ ] 已理解章节大纲的核心要点")
        prompt_parts.append(f"  [ ] 已明确本章要展现的核心冲突")
        prompt_parts.append(f"  [ ] 已熟悉本章涉及的人物特征")
        prompt_parts.append(f"  [ ] 已了解相关的世界观设定")
        prompt_parts.append(f"  [ ] 已确定写作风格和叙述语气")
        prompt_parts.append(f"  [ ] 已记住【本章完】标记和字数要求")
        prompt_parts.append("")
        prompt_parts.append("=" * 60)
        prompt_parts.append("现在请开始创作，确保符合以上所有要求。")
        prompt_parts.append("=" * 60)

        return "\n".join(prompt_parts)

    def _retrieve_relevant_worldview(
        self,
        chapter_outline: str,
        world_view: str,
        max_tokens: int
    ) -> str:
        """智能检索相关世界观片段"""
        if not world_view:
            return ""

        keywords = self._extract_keywords(chapter_outline)
        paragraphs = [p.strip() for p in world_view.split('\n\n') if p.strip()]

        scored_paragraphs = []
        for para in paragraphs:
            score = self._calculate_relevance(para, keywords)
            scored_paragraphs.append((score, para))

        scored_paragraphs.sort(key=lambda x: x[0], reverse=True)

        core_sections = ['法则', '地理', '历史', '势力', '社会结构', '世界观']
        for idx, (score, para) in enumerate(scored_paragraphs):
            for section in core_sections:
                if section in para:
                    scored_paragraphs[idx] = (score + 0.3, para)
                    break

        scored_paragraphs.sort(key=lambda x: x[0], reverse=True)

        selected_parts = []
        current_length = 0
        for score, para in scored_paragraphs:
            if current_length + len(para) <= max_tokens:
                selected_parts.append(para)
                current_length += len(para)
            else:
                remaining = max_tokens - current_length
                if remaining > 100:
                    selected_parts.append(para[:remaining] + '...')
                break

        return '\n\n'.join(selected_parts)

    def _retrieve_relevant_characters(
        self,
        chapter_outline: str,
        characters: List[Dict]
    ) -> List[str]:
        """智能检索相关人物"""
        if not characters:
            return []

        mentioned_names = self._extract_character_names(chapter_outline)

        relevant_chars = []
        for char in characters:
            basic_info = char.get('basic_info', {})
            name = basic_info.get('name', '')
            if not name:
                continue

            is_relevant = (
                name in mentioned_names or
                basic_info.get('role', '') in ['主角', '主要角色', '核心人物'] or
                len(relevant_chars) < 3
            )

            if is_relevant:
                compressed = self._compress_character(char)
                relevant_chars.append(compressed)

        return relevant_chars

    def _compress_style(self, style: str, max_tokens: int) -> str:
        """压缩风格描述"""
        if not style:
            return ""

        compressed_lines = []

        if '第一人称' in style or '第三人称' in style:
            perspective = '第一人称' if '第一人称' in style else '第三人称'
            compressed_lines.append(f"视角: {perspective}")

        style_keywords = {
            '文白': '半文半白',
            '文雅': '文雅优美',
            '通俗': '通俗平实',
            '幽默': '幽默风趣',
            '简洁': '简洁有力',
            '华丽': '华丽辞藻'
        }
        for key, value in style_keywords.items():
            if key in style:
                compressed_lines.append(f"语言: {value}")
                break

        if '快节奏' in style or '慢节奏' in style:
            pace = '快节奏' if '快节奏' in style else '慢节奏'
            compressed_lines.append(f"节奏: {pace}")

        if len(compressed_lines) < 3:
            paragraphs = style.split('\n\n')
            for para in paragraphs[:2]:
                if len(para) < max_tokens:
                    compressed_lines.append(para)

        result = '\n'.join(compressed_lines)

        if len(result) > max_tokens:
            result = result[:max_tokens] + '...'

        return result

    def _compress_character(self, char: Dict) -> str:
        """构建完整的人物档案卡片"""
        basic_info = char.get('basic_info', {})
        name = basic_info.get('name', '')
        if not name:
            return ""

        lines = []
        lines.append(f"\n【{name}】")

        # 基础信息
        basic_parts = []
        role = basic_info.get('role', '')
        if role:
            basic_parts.append(f"角色:{role}")

        age = basic_info.get('age', '')
        if age and age != '未知':
            basic_parts.append(f"年龄:{age}岁")

        gender = basic_info.get('gender', '')
        if gender and gender != '未知':
            basic_parts.append(f"性别:{gender}")

        mbti = basic_info.get('mbti', '')
        if mbti:
            basic_parts.append(f"MBTI:{mbti}")

        if basic_parts:
            lines.append("基础信息: " + " | ".join(basic_parts))

        # 性格特征
        personality = basic_info.get('personality', '')
        if personality:
            lines.append(f"性格特征: {personality}")

        # 外貌特征
        appearance = basic_info.get('appearance', '')
        if appearance:
            lines.append(f"外貌特征: {appearance}")

        # 背景故事
        background = basic_info.get('background', '')
        if background:
            if len(background) > 200:
                key_sentences = []
                sentences = background.replace('。', '。\n').split('\n')
                for sentence in sentences:
                    if any(keyword in sentence for keyword in ['经历', '过去', '曾经', '后来', '因为', '所以']):
                        key_sentences.append(sentence.strip())
                if key_sentences:
                    background = ''.join(key_sentences[:3])
                else:
                    background = background[:200] + '...'
            lines.append(f"背景故事: {background}")

        return '\n'.join(lines)

    def _extract_keywords(self, text: str) -> Set[str]:
        """从文本中提取关键词"""
        keywords = set()

        patterns = [
            r'([\u4e00-\u9fa5]{2,4})(城|山|河|岛|森林|宫殿|府|阁)',
            r'([\u4e00-\u9fa5]{2,4})(派|门|宗|教|国|朝|军)',
            r'([\u4e00-\u9fa5]{2,4})(剑|刀|法|术|功|诀)',
        ]

        for pattern in patterns:
            matches = re.findall(pattern, text)
            for match in matches:
                keywords.add(match[0])

        return keywords

    def _extract_character_names(self, text: str) -> Set[str]:
        """从文本中提取人物名"""
        names = re.findall(r'([\u4e00-\u9fa5]{2,3})', text)

        stop_words = {'世界', '中国', '大陆', '森林', '山脉', '河流', '宫殿',
                     '国家', '军队', '人民', '百姓', '江湖', '武林', '修仙'}

        return set(name for name in names if name not in stop_words)

    def _calculate_relevance(self, paragraph: str, keywords: Set[str]) -> float:
        """计算段落与关键词的相关性评分"""
        if not keywords:
            return 0.5

        score = 0.0
        for keyword in keywords:
            if keyword in paragraph:
                score += 1.0

        return score / len(keywords)

    def _build_context_summary(self, previous_chapters: List[str]) -> str:
        """构建前情提要摘要"""
        if not previous_chapters:
            return ""

        recent = previous_chapters[-2:]

        summaries = []
        for i, chapter in enumerate(recent, 1):
            summary = chapter[:200] + ('...' if len(chapter) > 200 else '')
            summaries.append(f"第{i}段: {summary}")

        return '\n'.join(summaries)

    def _extract_conflict_hooks(self, chapter_outline: str) -> List[str]:
        """从章节大纲中提取冲突钩子"""
        conflict_hooks = []

        conflict_patterns = {
            '人与人矛盾': ['对峙', '冲突', '对抗', '争论', '争斗', '竞争', '敌对', '仇恨', '复仇'],
            '人物与命运': ['困境', '危机', '挑战', '遭遇', '面对', '逃脱', '生存', '考验', '磨难'],
            '内心矛盾': ['矛盾', '挣扎', '犹豫', '抉择', '困惑', '觉醒', '成长', '改变', '决心'],
            '悬念未知': ['发现', '揭露', '秘密', '真相', '谜团', '意外', '转折', '突变', '震惊']
        }

        for conflict_type, keywords in conflict_patterns.items():
            for keyword in keywords:
                if keyword in chapter_outline:
                    sentences = chapter_outline.split('。')
                    for sentence in sentences:
                        if keyword in sentence:
                            hook = sentence.strip()[:50]
                            if hook and hook not in conflict_hooks:
                                conflict_hooks.append(hook)

        if not conflict_hooks:
            conflict_hooks.append("展现本章情节中的核心冲突和矛盾")
            conflict_hooks.append("构建必要的张力和悬念")

        return conflict_hooks[:3]
```
