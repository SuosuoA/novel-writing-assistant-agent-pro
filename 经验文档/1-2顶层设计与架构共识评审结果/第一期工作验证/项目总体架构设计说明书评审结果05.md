```python
class UIComponent(ABC):
    """UI组件基类 - 统一组件生命周期管理"""
    
    def __init__(self, parent: tk.Widget, context: PluginContext):
        self.parent = parent
        self.context = context
        self._frame: Optional[tk.Frame] = None
        self._event_subscriptions: List[str] = []
    
    @abstractmethod
    def create_ui(self) -> tk.Widget:
        """创建UI组件，返回根Widget"""
        pass
    
    def subscribe_event(self, event_type: str, handler: Callable):
        """订阅事件（自动管理订阅ID）"""
        sub_id = self.context.event_bus.subscribe(event_type, handler)
        self._event_subscriptions.append(sub_id)
    
    def destroy(self):
        """销毁组件时清理事件订阅"""
        for sub_id in self._event_subscriptions:
            self.context.event_bus.unsubscribe(sub_id)
        self._event_subscriptions.clear()
```

```python
# 建议补充：UI状态管理器
class UIStateManager:
    """UI状态管理器 - 统一管理UI状态"""
    
    def __init__(self, event_bus: EventBus):
        self._state: Dict[str, Any] = {}
        self._listeners: Dict[str, List[Callable]] = {}
        self._event_bus = event_bus
    
    def set_state(self, key: str, value: Any):
        """设置状态并通知监听者"""
        old_value = self._state.get(key)
        self._state[key] = value
        
        # 通知本地监听者
        for listener in self._listeners.get(key, []):
            listener(value, old_value)
        
        # 发布全局状态变更事件
        self._event_bus.publish("ui.state_changed", {
            "key": key,
            "value": value,
            "old_value": old_value
        })
```

### 2.2 Tkinter线程安全规范审查

#### 现状分析

文档第2.1节已定义线程安全规范：
- 所有UI操作必须在主线程执行
- 异步事件的handler在独立线程池执行，禁止直接操作UI
- 需要更新UI时，使用`widget.after()`调度到主线程
- `publish_sync()`在调用线程执行，主线程调用时可直接操作UI

#### 评审意见

**✅ 规范定义正确**，但建议补充**代码示例**和**错误检测机制**：

```python
# 建议补充：线程安全检查装饰器
import threading
import functools

def ui_thread_only(func):
    """确保方法只在UI线程执行的装饰器"""
    @functools.wraps(func)
    def wrapper(self, *args, **kwargs):
        if threading.current_thread() is not threading.main_thread():
            # 自动调度到主线程
            if hasattr(self, 'root') and self.root:
                self.root.after(0, lambda: func(self, *args, **kwargs))
                return
            else:
                raise RuntimeError(
                    f"UI方法 {func.__name__} 必须在主线程调用。"
                    f"当前线程: {threading.current_thread().name}"
                )
        return func(self, *args, **kwargs)
    return wrapper

# 使用示例
class MainWindow:
    @ui_thread_only
    def update_status(self, text: str):
        """更新状态栏（强制主线程执行）"""
        self.status_label.config(text=text)
```

### 6.1 Tkinter最佳实践

```python
# Tkinter线程安全最佳实践示例
import tkinter as tk
from tkinter import ttk
import threading
import queue

class ThreadSafeUI:
    """线程安全UI更新示例"""
    
    def __init__(self, root):
        self.root = root
        self.update_queue = queue.Queue()
        self._check_queue()
    
    def _check_queue(self):
        """定期检查更新队列"""
        try:
            while True:
                func = self.update_queue.get_nowait()
                func()
        except queue.Empty:
            pass
        finally:
            self.root.after(100, self._check_queue)
    
    def update_ui_from_thread(self, func):
        """从工作线程安全更新UI"""
        self.update_queue.put(func)
```
