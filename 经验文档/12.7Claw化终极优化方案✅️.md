def check_data_completeness(self) -> Dict[str, Any]:
    """检查知识库数据完整性"""
    required = {
        "scifi": 559,
        "xuanhuan": 334,
        "general": 237
    }
    
    actual = {}
    for category, target in required.items():
        data_path = self.data_dir / "knowledge" / category
        if data_path.exists():
            actual[category] = sum(
                len(json.load(f).get("knowledge_points", []))
                for f in data_path.glob("*.json")
            )
        else:
            actual[category] = 0
    
    completion = {
        cat: {
            "target": target,
            "actual": actual.get(cat, 0),
            "completion_rate": actual.get(cat, 0) / target
        }
        for cat, target in required.items()
    }
    
    overall_rate = sum(c["completion_rate"] for c in completion.values()) / len(completion)
    
    return {
        "categories": completion,
        "overall_completion_rate": overall_rate,
        "status": "complete" if overall_rate >= 0.95 else "incomplete"
    }
```

class RealtimeMetricsMonitor:
    """实时指标监控面板"""
    
    def __init__(self, workspace: Path):
        self.workspace = workspace
        self.metrics_file = workspace / "data" / "realtime_metrics.json"
        self._init_storage()
    
    def update_metric(self, chapter_id: str, dimension: str, value: float):
        """更新单个指标"""
        timestamp = datetime.now().isoformat()
        
        metric_data = {
            "chapter_id": chapter_id,
            "dimension": dimension,
            "value": value,
            "timestamp": timestamp
        }
        
        # 追加到文件
        with open(self.metrics_file, 'a', encoding='utf-8') as f:
            f.write(json.dumps(metric_data, ensure_ascii=False) + '\n')
    
    def update_batch(self, chapter_id: str, metrics: Dict[str, float]):
        """批量更新指标"""
        for dimension, value in metrics.items():
            self.update_metric(chapter_id, dimension, value)
    
    def get_current_metrics(self, window_minutes: int = 60) -> Dict[str, Any]:
        """获取最近N分钟的指标"""
        cutoff_time = datetime.now() - timedelta(minutes=window_minutes)
        
        recent_metrics = []
        with open(self.metrics_file, 'r', encoding='utf-8') as f:
            for line in f:
                metric = json.loads(line)
                if metric["timestamp"] >= cutoff_time.isoformat():
                    recent_metrics.append(metric)
        
        # 计算实时统计
        by_dimension = {}
        for metric in recent_metrics:
            dim = metric["dimension"]
            if dim not in by_dimension:
                by_dimension[dim] = []
            by_dimension[dim].append(metric["value"])
        
        return {
            "window_minutes": window_minutes,
            "total_chapters": len(recent_metrics),
            "dimension_stats": {
                dim: {
                    "avg": sum(values) / len(values),
                    "min": min(values),
                    "max": max(values),
                    "count": len(values)
                }
                for dim, values in by_dimension.items()
            }
        }

class KnowledgeGraph:
    """知识库知识点关联网络"""
    
    def __init__(self, workspace: Path):
        self.workspace = workspace
        self.graph_file = workspace / "data" / "knowledge_graph.json"
        self._load_graph()
    
    def build_relationships(self, similarity_threshold: float = 0.3) -> int:
        """构建知识点关联关系"""
        all_knowledge = self._load_all_knowledge()
        
        # 使用AI自动识别关联
        relationships = []
        for i, kp1 in enumerate(all_knowledge):
            for kp2 in all_knowledge[i+1:]:
                # 判断是否相关
                similarity = self._calculate_similarity(kp1, kp2)
                
                if similarity > similarity_threshold:
                    relationships.append({
                        "source": kp1.knowledge_id,
                        "target": kp2.knowledge_id,
                        "type": "related",
                        "weight": similarity
                    })
        
        self.graph_data["relationships"] = relationships
        self._save_graph()
        
        return len(relationships)
    
    def get_related_knowledge(self, knowledge_id: str, top_k: int = 5) -> List[Dict]:
        """获取关联知识点"""
        related = []
        for rel in self.graph_data.get("relationships", []):
            if rel["source"] == knowledge_id:
                related.append({
                    "knowledge_id": rel["target"],
                    "similarity": rel["weight"]
                })
        
        # 按相似度排序
        related.sort(key=lambda x: x["similarity"], reverse=True)
        return related[:top_k]
    
    def get_cluster_stats(self) -> Dict[str, Any]:
        """获取知识图谱统计信息"""
        all_nodes = set()
        for rel in self.graph_data.get("relationships", []):
            all_nodes.add(rel["source"])
            all_nodes.add(rel["target"])
        
        return {
            "total_nodes": len(all_nodes),
            "total_relations": len(self.graph_data.get("relationships", [])),
            "avg_relations_per_node": len(self.graph_data.get("relationships", [])) * 2 / len(all_nodes) if all_nodes else 0
        }
```

class SentimentAnalyzer:

        """分析反馈文本的情感倾向"""

分析以下用户反馈的情感倾向,输出JSON:

反馈内容: {feedback_text}

输出格式:
{{
  "sentiment": "positive|negative|neutral",
  "confidence": 0.95,
  "key_issues": ["风格", "连贯性"],
  "intensity": 0.8
}}

sentiment: 情感倾向(正面/负面/中性)
confidence: 置信度(0-1)
key_issues: 提及的关键问题(最多3个)
intensity: 情感强度(0-1)
"""
        
        result = self._call_ai(sentiment_prompt)
        analysis = json.loads(result)
        
        # 添加建议的反馈类型
        analysis["suggested_type"] = analysis["sentiment"]
        
        return analysis

def collect_feedback(self, chapter_id: str, details: str, rating: Optional[float] = None):

    analyzer = SentimentAnalyzer(self.workspace)
