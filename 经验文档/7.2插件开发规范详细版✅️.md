## 一、规范概述

### 1.1 适用范围

本规范适用于所有Agent Pro插件的开发，包括：
- 分析器插件（Analyzer）
- 生成器插件（Generator）
- 验证器插件（Validator）
- 工具插件（Tool）
- 协议插件（Protocol）
- AI插件（AI）
- 存储插件（Storage）

### 1.2 核心原则

| 原则 | 说明 |
|------|------|
| **单一职责** | 每个插件只负责一个明确的功能 |
| **显式依赖** | 所有依赖在plugin.json中声明 |
| **优雅降级** | 依赖不可用时提供基本功能或明确提示 |
| **资源隔离** | 插件资源独立管理，不泄漏到外部 |
| **线程安全** | 共享状态必须加锁保护 |

### 1.3 技术栈要求

- **Python版本**: 3.12.x
- **数据验证**: Pydantic v2
- **日志模块**: 使用PluginContext提供的logger
- **异步支持**: 使用asyncio（如需要）

## 二、目录结构规范

### 2.1 标准目录结构

```
plugins/
└── {plugin-id}/                    # 插件根目录（如：outline-parser-v3）
    ├── plugin.json                 # 插件清单（必需）
    ├── __init__.py                 # Python包初始化（必需）
    ├── main.py                     # 插件主入口类（必需）
    ├── config.py                   # 配置模型定义（推荐）
    ├── models.py                   # 数据模型定义（推荐）
    ├── utils.py                    # 工具函数（可选）
    ├── constants.py                # 常量定义（可选）
    ├── exceptions.py               # 自定义异常（可选）
    ├── tests/                      # 单元测试目录（推荐）
    │   ├── __init__.py
    │   └── test_main.py
    ├── schemas/                    # JSON Schema文件（可选）
    │   └── config.schema.json
    ├── templates/                  # 模板文件（可选）
    │   └── prompt_template.yaml
    ├── resources/                  # 资源文件（可选）
    │   └── icons/
    └── README.md                   # 插件说明文档（必需）
```

### 2.2 目录命名规范

| 规则 | 说明 | 示例 |
|------|------|------|
| 插件ID | 小写字母+数字+连字符 | `outline-parser-v3` |
| 版本后缀 | 重大版本号 | `-v1`, `-v2`, `-v3` |
| Python文件 | 小写+下划线 | `context_builder.py` |
| 测试文件 | test_前缀 | `test_main.py` |

### 2.3 文件大小限制

| 文件类型 | 大小限制 | 说明 |
|---------|---------|------|
| 单个.py文件 | ≤ 1000行 | 超过应拆分模块 |
| plugin.json | ≤ 10KB | 清单文件保持简洁 |
| README.md | ≤ 50KB | 文档不宜过长 |
| 资源文件 | ≤ 5MB/个 | 大文件应外部托管 |

### 4.2 异常处理模式

```python
# main.py

from typing import Optional, Tuple
from .exceptions import PluginError, PluginExecutionError

class MyPlugin(BasePlugin):
    
    def execute_with_error_handling(self, data: dict) -> Tuple[bool, Optional[str]]:
        """带错误处理的执行方法
        
        Returns:
            Tuple[bool, Optional[str]]: (成功标志, 错误信息)
        """
        try:
            # 1. 参数验证
            if not data:
                return False, "输入数据不能为空"
            
            # 2. 依赖检查
            if not self._required_service:
                return False, "必需服务不可用"
            
            # 3. 执行核心逻辑
            result = self._execute_core(data)
            
            # 4. 结果验证
            if not self._validate_result(result):
                return False, "执行结果验证失败"
            
            return True, None
            
        except PluginError as e:
            # 已知插件错误
            self._logger.error(f"插件执行失败: {e}", exc_info=True)
            return False, str(e)
            
        except Exception as e:
            # 未知错误
            self._logger.error(f"未知错误: {e}", exc_info=True)
            return False, f"内部错误: {type(e).__name__}"
    
    def _execute_core(self, data: dict) -> dict:
        """核心逻辑"""
        pass
    
    def _validate_result(self, result: dict) -> bool:
        """结果验证"""
        return result is not None and "success" in result
```

### 4.3 错误恢复策略

```python
class ResilientPlugin(BasePlugin):
    """支持错误恢复的插件"""
    
    def __init__(self, context: PluginContext):
        super().__init__(context)
        self._fallback_enabled = True
        self._max_retries = 3
    
    def execute_with_fallback(self, data: dict) -> dict:
        """带降级策略的执行"""
        
        # 尝试主要方法
        for attempt in range(self._max_retries):
            try:
                return self._execute_primary(data)
            except Exception as e:
                self._logger.warning(
                    f"主要方法执行失败(尝试 {attempt + 1}/{self._max_retries}): {e}"
                )
                if attempt < self._max_retries - 1:
                    continue
        
        # 主要方法失败，尝试降级方法
        if self._fallback_enabled:
            self._logger.info("切换到降级方法")
            try:
                return self._execute_fallback(data)
            except Exception as e:
                self._logger.error(f"降级方法也失败: {e}")
        
        # 全部失败
        return {
            "success": False,
            "error": "所有执行路径都失败",
            "fallback_attempted": self._fallback_enabled
        }
    
    def _execute_primary(self, data: dict) -> dict:
        """主要执行方法"""
        pass
    
    def _execute_fallback(self, data: dict) -> dict:
        """降级执行方法"""
        pass
```

### 5.2 配置读取模式

```python
# main.py

class ConfigurablePlugin(BasePlugin):
    """支持配置的插件"""
    
    def __init__(self, context: PluginContext):
        super().__init__(context)
        self._config: Optional[PluginConfig] = None
    
    def initialize(self, context: PluginContext) -> bool:
        """初始化时加载配置"""
        try:
            # 从上下文读取配置
            config_data = self._load_config(context)
            
            # 验证并创建配置对象
            self._config = PluginConfig(**config_data)
            
            self._logger.info(
                f"[{self._metadata.id}] 配置加载成功",
                extra={"config": self._config.model_dump()}
            )
            return True
            
        except Exception as e:
            self._logger.error(f"配置加载失败: {e}", exc_info=True)
            # 使用默认配置
            self._config = PluginConfig()
            return True
    
    def _load_config(self, context: PluginContext) -> dict:
        """加载配置"""
        # 优先级：上下文配置 > plugin.json配置 > 默认值
        
        # 1. 从上下文获取配置
        if hasattr(context, 'config') and context.config:
            return context.config
        
        # 2. 从配置管理器获取
        if self._config_manager:
            config_key = f"plugins.{self._metadata.id}"
            return self._config_manager.get(config_key, {})
        
        # 3. 返回空字典，使用默认值
        return {}
    
    def set_config(self, config: dict) -> bool:
        """运行时设置配置"""
        try:
            self._config = PluginConfig(**config)
            self._logger.info(f"配置更新成功: {self._config.model_dump()}")
            return True
        except Exception as e:
            self._logger.error(f"配置更新失败: {e}")
            return False
    
    def get_config(self) -> dict:
        """获取当前配置"""
        return self._config.model_dump() if self._config else {}
```

## 七、资源管理规范

### 7.1 资源生命周期

```python
class ResourceAwarePlugin(BasePlugin):
    """资源感知插件"""
    
    def __init__(self, context: PluginContext):
        super().__init__(context)
        
        # 资源声明
        self._cache: dict = {}
        self._history: list = []
        self._thread_pool: Optional[ThreadPoolExecutor] = None
        self._temp_files: list[str] = []
    
    def initialize(self, context: PluginContext) -> bool:
        """初始化资源"""
        try:
            # 创建线程池
            self._thread_pool = ThreadPoolExecutor(max_workers=3)
            
            # 加载缓存
            self._load_cache()
            
            return True
        except Exception as e:
            self._logger.error(f"资源初始化失败: {e}")
            return False
    
    def shutdown(self) -> None:
        """释放所有资源"""
        self._logger.info(f"[{self._metadata.id}] 开始释放资源...")
        
        # 1. 关闭线程池
        if self._thread_pool:
            self._thread_pool.shutdown(wait=False)
            self._thread_pool = None
        
        # 2. 清理缓存
        self._cache.clear()
        
        # 3. 清理历史记录
        self._history.clear()
        
        # 4. 删除临时文件
        for temp_file in self._temp_files:
            try:
                Path(temp_file).unlink(missing_ok=True)
            except Exception as e:
                self._logger.warning(f"删除临时文件失败: {temp_file}, {e}")
        self._temp_files.clear()
        
        # 5. 调用父类shutdown
        super().shutdown()
        
        self._logger.info(f"[{self._metadata.id}] 资源释放完成")
```

### 7.2 内存管理

```python
class MemoryManagedPlugin(BasePlugin):
    """内存管理插件"""
    
    # 历史记录限制
    MAX_HISTORY_SIZE = 50
    TRUNCATE_TO_SIZE = 30
    
    # 缓存限制
    MAX_CACHE_SIZE = 1000
    
    def _add_to_history(self, item: dict) -> None:
        """添加到历史记录（自动限制大小）"""
        self._history.append(item)
        
        # 超过限制时截断
        if len(self._history) > self.MAX_HISTORY_SIZE:
            self._history = self._history[-self.TRUNCATE_TO_SIZE:]
            self._logger.debug(f"历史记录已截断至{self.TRUNCATE_TO_SIZE}条")
    
    def _cache_set(self, key: str, value: Any) -> None:
        """设置缓存（LRU淘汰）"""
        if len(self._cache) >= self.MAX_CACHE_SIZE:
            # 删除最旧的条目
            oldest_key = next(iter(self._cache))
            del self._cache[oldest_key]
            self._logger.debug(f"缓存已满，淘汰: {oldest_key}")
        
        self._cache[key] = value
    
    def clear_cache(self) -> None:
        """清理缓存"""
        self._cache.clear()
        self._logger.info(f"[{self._metadata.id}] 缓存已清理")
```

## 八、线程安全规范

### 8.1 共享状态保护

```python
import threading
from typing import Dict, Any

class ThreadSafePlugin(BasePlugin):
    """线程安全插件"""
    
    def __init__(self, context: PluginContext):
        super().__init__(context)
        
        # 使用RLock保护共享状态
        self._lock = threading.RLock()
        self._state: Dict[str, Any] = {}
        self._counter = 0
    
    def get_state(self, key: str) -> Any:
        """线程安全读取状态"""
        with self._lock:
            return self._state.get(key)
    
    def set_state(self, key: str, value: Any) -> None:
        """线程安全设置状态"""
        with self._lock:
            self._state[key] = value
    
    def increment_counter(self) -> int:
        """线程安全计数器"""
        with self._lock:
            self._counter += 1
            return self._counter
    
    def atomic_operation(self, data: dict) -> dict:
        """原子操作示例"""
        with self._lock:
            # 读取-修改-写入 原子操作
            current = self._state.get("value", 0)
            new_value = current + data.get("increment", 1)
            self._state["value"] = new_value
            return {"value": new_value}
```

### 8.2 避免死锁

```python
class DeadlockFreePlugin(BasePlugin):
    """避免死锁的插件"""
    
    def __init__(self, context: PluginContext):
        super().__init__(context)
        self._lock_a = threading.RLock()
        self._lock_b = threading.RLock()
    
    def operation_a_then_b(self) -> None:
        """正确：按固定顺序获取锁"""
        with self._lock_a:  # 先获取A
            with self._lock_b:  # 再获取B
                # 执行操作
                pass
    
    def operation_b_then_a(self) -> None:
        """错误：相反顺序获取锁，可能死锁"""
        # ❌ 不要这样做！
        # with self._lock_b:
        #     with self._lock_a:
        #         pass
        
        # ✅ 正确：保持相同顺序
        with self._lock_a:
            with self._lock_b:
                pass
    
    def safe_cross_plugin_call(self, other_plugin: 'DeadlockFreePlugin') -> None:
        """跨插件调用安全模式"""
        # 使用超时锁
        if self._lock_a.acquire(timeout=5.0):
            try:
                # 执行需要锁A的操作
                pass
            finally:
                self._lock_a.release()
        else:
            self._logger.warning("获取锁超时，放弃操作")

@dataclass
class ConsistencyReport:
    """一致性分析报告"""
    
    report_id: str                      # 报告唯一标识
    project_name: str                   # 项目名称
    chapters_analyzed: int              # 分析的章节数量
    issues: List[ConsistencyIssue]      # 冲突项列表
    summary: str                        # 分析摘要
    analyzed_at: float                  # 分析时间戳
    high_priority_count: int = 0        # 高优先级冲突数量
    medium_priority_count: int = 0      # 中等优先级冲突数量
    low_priority_count: int = 0         # 低优先级冲突数量
    
    def __post_init__(self):
        if not self.report_id:
            self.report_id = f"report-{uuid.uuid4().hex[:8]}"
        if not self.analyzed_at:
            self.analyzed_at = time.time()
        self._update_counts()
    
    def add_issue(self, issue: ConsistencyIssue) -> None:
        """添加冲突项并自动更新统计"""
        self.issues.append(issue)
        self._update_counts()
    
    def _update_counts(self) -> None:
        """更新统计数量"""
        self.high_priority_count = sum(
            1 for i in self.issues if i.severity == ConsistencySeverity.HIGH
        )
        self.medium_priority_count = sum(
            1 for i in self.issues if i.severity == ConsistencySeverity.MEDIUM
        )
        self.low_priority_count = sum(
            1 for i in self.issues if i.severity == ConsistencySeverity.LOW
        )
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        return {
            "report_id": self.report_id,
            "project_name": self.project_name,
            "chapters_analyzed": self.chapters_analyzed,
            "issues": [i.to_dict() for i in self.issues],
            "summary": self.summary,
            "analyzed_at": self.analyzed_at,
            "high_priority_count": self.high_priority_count,
            "medium_priority_count": self.medium_priority_count,
            "low_priority_count": self.low_priority_count
        }

    base_score = 1.0

        report.high_priority_count * 0.20 +
        report.medium_priority_count * 0.10 +
        report.low_priority_count * 0.05
    )
    return max(0.4, base_score - penalty)
```
