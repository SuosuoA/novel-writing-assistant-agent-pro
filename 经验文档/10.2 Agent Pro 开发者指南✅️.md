## 三、插件开发基础

### 3.1 插件类型

| 类型 | 接口类 | 说明 |
|------|--------|------|
| analyzer | AnalyzerPlugin | 分析器插件（大纲解析、风格学习等） |
| generator | GeneratorPlugin | 生成器插件（内容生成） |
| validator | ValidatorPlugin | 验证器插件（质量验证） |
| tool | ToolPlugin | 工具插件（通用功能） |
| protocol | ProtocolPlugin | 协议插件（导入导出） |
| ai | AIPlugin | AI插件（LLM调用封装） |
| storage | StoragePlugin | 存储插件（数据持久化） |

### 3.2 插件生命周期

```
UNLOADED → LOADED → ACTIVE → ERROR → UNLOADING
```

| 状态 | 说明 |
|------|------|
| UNLOADED | 未加载（初始状态） |
| LOADED | 已加载，未激活 |
| ACTIVE | 已激活，可执行 |
| ERROR | 错误状态 |
| UNLOADING | 卸载中 |

### 3.3 必需文件

每个插件必须包含以下文件：

```
plugins/my-plugin/
├── plugin.json    # 插件清单（必需）
├── __init__.py    # 包入口（必需）
├── plugin.py      # 插件主类（推荐）
└── README.md      # 说明文档（推荐）
```

### 4.1 分析器插件示例

```python
# plugins/my-analyzer/plugin.py

from core.plugin_interface import (
    AnalyzerPlugin, PluginMetadata, PluginType, PluginContext
)
from typing import Dict, Any, List, Optional

class MyAnalyzerPlugin(AnalyzerPlugin):
    """自定义分析器插件"""
    
    PLUGIN_ID = "my-analyzer-v1"
    PLUGIN_VERSION = "1.0.0"
    
    def __init__(self):
        super().__init__()
        self._cache: Dict[str, Any] = {}
    
    @classmethod
    def get_metadata(cls) -> PluginMetadata:
        return PluginMetadata(
            id=cls.PLUGIN_ID,
            name="我的分析器",
            version=cls.PLUGIN_VERSION,
            description="自定义内容分析器",
            author="开发者",
            plugin_type=PluginType.ANALYZER
        )
    
    def initialize(self, context: PluginContext) -> bool:
        if not super().initialize(context):
            return False
        
        # 订阅事件
        self._context.event_bus.subscribe(
            "content.analyze_requested",
            self._on_analyze_requested
        )
        
        return True
    
    def analyze(self, content: str, 
                options: Optional[Dict[str, Any]] = None) -> Dict[str, Any]:
        """分析内容"""
        options = options or {}
        
        # 实现分析逻辑
        result = {
            "word_count": len(content),
            "line_count": content.count('\n') + 1,
            "analysis_type": options.get("type", "basic")
        }
        
        return result
    
    def get_supported_formats(self) -> List[str]:
        return ["txt", "md", "json"]
    
    def get_analysis_types(self) -> List[str]:
        return ["basic", "detailed", "statistical"]
    
    def _on_analyze_requested(self, event):
        """事件处理"""
        content = event.data.get("content", "")
        result = self.analyze(content)
        
        self._context.event_bus.publish(
            "content.analyze_completed",
            {"result": result}
        )
    
    def shutdown(self) -> bool:
        """关闭插件"""
        if self._context:
            self._context.event_bus.unsubscribe("content.analyze_requested")
        self._cache.clear()
        return True
```

```python
from core.plugin_interface import (
    GeneratorPlugin, PluginMetadata, PluginType, PluginContext
)
from core.models import GenerationRequest, GenerationResult
from typing import Dict, Any, List, Tuple

class MyGeneratorPlugin(GeneratorPlugin):
    """自定义生成器插件"""
    
    PLUGIN_ID = "my-generator-v1"
    PLUGIN_VERSION = "1.0.0"
    
    def __init__(self):
        super().__init__()
        self._llm_client = None
    
    @classmethod
    def get_metadata(cls) -> PluginMetadata:
        return PluginMetadata(
            id=cls.PLUGIN_ID,
            name="我的生成器",
            version=cls.PLUGIN_VERSION,
            description="自定义内容生成器",
            author="开发者",
            plugin_type=PluginType.GENERATOR
        )
    
    def initialize(self, context: PluginContext) -> bool:
        if not super().initialize(context):
            return False
        
        # 获取LLM客户端
        self._llm_client = context.service_locator.get("llm_client")
        
        return True
    
    def generate(self, request: GenerationRequest) -> GenerationResult:
        """生成内容"""
        # 构建Prompt
        prompt = self._build_prompt(request)
        
        # 调用LLM
        content = self._call_llm(prompt)
        
        return GenerationResult(
            request_id=request.request_id,
            content=content,
            word_count=len(content),
            iteration_count=1,
            validation_scores=None
        )
    
    def validate_request(self, request: GenerationRequest) -> Tuple[bool, List[str]]:
        """验证请求"""
        errors = []
        
        if not request.title:
            errors.append("标题不能为空")
        if request.word_count < 100:
            errors.append("字数不能少于100")
        
        return len(errors) == 0, errors
    
    def get_generation_options(self) -> Dict[str, Any]:
        """获取生成选项"""
        return {
            "style": {
                "type": "string",
                "enum": ["formal", "casual", "poetic"],
                "default": "formal"
            },
            "length": {
                "type": "integer",
                "default": 1000,
                "min": 100,
                "max": 5000
            }
        }
    
    def _build_prompt(self, request: GenerationRequest) -> str:
        """构建Prompt"""
        return f"""请根据以下大纲生成小说章节：

标题：{request.title}
大纲：{request.outline}
目标字数：{request.word_count}

请直接输出正文内容。"""
    
    def _call_llm(self, prompt: str) -> str:
        """调用LLM"""
        if self._llm_client:
            return self._llm_client.call(prompt)
        return "LLM客户端未配置"
```

### 4.3 验证器插件示例

```python
# plugins/my-validator/plugin.py

from core.plugin_interface import (
    ValidatorPlugin, PluginMetadata, PluginType, PluginContext
)
from core.models import ValidationScores
from typing import Dict, Any, List, Optional

class MyValidatorPlugin(ValidatorPlugin):
    """自定义验证器插件"""
    
    PLUGIN_ID = "my-validator-v1"
    PLUGIN_VERSION = "1.0.0"
    
    @classmethod
    def get_metadata(cls) -> PluginMetadata:
        return PluginMetadata(
            id=cls.PLUGIN_ID,
            name="我的验证器",
            version=cls.PLUGIN_VERSION,
            description="自定义内容验证器",
            author="开发者",
            plugin_type=PluginType.VALIDATOR
        )
    
    def validate(self, content: str, 
                 context: Optional[Dict[str, Any]] = None) -> ValidationScores:
        """验证内容"""
        scores = ValidationScores()
        
        # 字数评分
        word_count = len(content)
        target = context.get("target_word_count", 2000) if context else 2000
        scores.word_count_score = min(1.0, word_count / target)
        
        # 自然度评分（简化示例）
        scores.naturalness_score = self._evaluate_naturalness(content)
        
        # 检查章节结束标记
        scores.has_chapter_end = "【本章完】" in content
        
        # 计算总分
        scores.calculate_total()
        
        return scores
    
    def get_validation_dimensions(self) -> List[str]:
        return ["word_count", "naturalness", "chapter_end"]
    
    def _evaluate_naturalness(self, content: str) -> float:
        """评估自然度"""
        # 简化实现
        sentences = content.count('。') + content.count('！') + content.count('？')
        if sentences == 0:
            return 0.5
        
        avg_length = len(content) / sentences
        
        # 平均句长在15-50字之间为最佳
        if 15 <= avg_length <= 50:
            return 0.9
        elif 10 <= avg_length <= 80:
            return 0.7
        else:
            return 0.5
```

```python
    TWIST = "twist"                  # 转折揭示

class ContinuationVersion(BaseModel):
    """续写版本"""
    version_id: str
    content: str
    word_count: int
    direction: str
    temperature: float
    created_at: datetime = Field(default_factory=datetime.now)
    score: Optional[float] = None

class ContinuationPlugin(GeneratorPlugin):
    """续写生成器插件"""
    
    PLUGIN_ID = "continuation-generator"
    PLUGIN_VERSION = "1.0.0"
    
    MAX_VERSIONS = 5
    DIRECTIONS = [
        ContinuationDirection.NATURAL,
        ContinuationDirection.PLOT_ADVANCE,
        ContinuationDirection.CLIMAX_SETUP,
        ContinuationDirection.CHARACTER_DEEP,
        ContinuationDirection.DIALOGUE,
        ContinuationDirection.SCENE,
        ContinuationDirection.SUSPENSE,
        ContinuationDirection.CONFLICT,
        ContinuationDirection.EMOTION,
        ContinuationDirection.TWIST
    ]
    
    def __init__(self):
        super().__init__()
        self._llm_client = None
        self._versions: List[ContinuationVersion] = []
    
    @classmethod
    def get_metadata(cls) -> PluginMetadata:
        return PluginMetadata(
            id=cls.PLUGIN_ID,
            name="续写生成器",
            version=cls.PLUGIN_VERSION,
            description="基于起始文本续写小说",
            author="Agent Pro Team",
            plugin_type=PluginType.GENERATOR
        )
    
    def initialize(self, context: PluginContext) -> bool:
        if not super().initialize(context):
            return False
        
        self._llm_client = context.service_locator.get("llm_client")
        return True
    
    def generate(self, request: GenerationRequest) -> GenerationResult:
        """生成续写内容"""
        # 从请求中提取参数
        starting_text = request.outline  # 复用outline字段
        word_count = request.word_count
        direction = request.title  # 复用title字段传递方向
        temperature = 0.8
        
        # 构建Prompt
        prompt = self._build_continuation_prompt(
            starting_text, word_count, direction
        )
        
        # 调用LLM
        content = self._call_llm(prompt, temperature)
        
        # 创建版本
        version = ContinuationVersion(
            version_id=f"v{len(self._versions) + 1}",
            content=content,
            word_count=len(content),
            direction=direction,
            temperature=temperature
        )
        
        # 添加到版本列表
        if len(self._versions) >= self.MAX_VERSIONS:
            self._versions.pop(0)
        self._versions.append(version)
        
        return GenerationResult(
            request_id=request.request_id,
            content=content,
            word_count=len(content),
            iteration_count=1,
            validation_scores=None
        )
    
    def generate_multiple_versions(
        self,
        starting_text: str,
        word_count: int,
        direction: str,
        num_versions: int = 3
    ) -> List[ContinuationVersion]:
        """生成多个版本"""
        versions = []
        temperatures = [0.6, 0.8, 1.0]
        
        for i in range(min(num_versions, len(temperatures))):
            temp = temperatures[i]
            prompt = self._build_continuation_prompt(
                starting_text, word_count, direction
            )
            content = self._call_llm(prompt, temp)
            
            version = ContinuationVersion(
                version_id=f"v{len(self._versions) + i + 1}",
                content=content,
                word_count=len(content),
                direction=direction,
                temperature=temp
            )
            versions.append(version)
        
        return versions
    
    def select_best_version(
        self, versions: List[ContinuationVersion]
    ) -> Tuple[ContinuationVersion, int]:
        """选择最佳版本"""
        if not versions:
            raise ValueError("版本列表为空")
        
        # 评分逻辑（简化）
        for v in versions:
            score = self._calculate_score(v)
            v.score = score
        
        # 选择评分最高的
        best_idx = max(range(len(versions)), key=lambda i: versions[i].score)
        return versions[best_idx], best_idx
    
    def validate_request(self, request: GenerationRequest) -> Tuple[bool, List[str]]:
        """验证请求"""
        errors = []
        
        if not request.outline:  # starting_text
            errors.append("起始文本不能为空")
        if request.word_count < 100 or request.word_count > 5000:
            errors.append("字数范围应为100-5000")
        
        return len(errors) == 0, errors
    
    def get_generation_options(self) -> Dict[str, Any]:
        """获取生成选项"""
        return {
            "direction": {
                "type": "string",
                "enum": self.DIRECTIONS,
                "default": ContinuationDirection.NATURAL
            },
            "word_count": {
                "type": "integer",
                "default": 1000,
                "min": 100,
                "max": 5000
            },
            "temperature": {
                "type": "number",
                "default": 0.8,
                "min": 0.0,
                "max": 2.0
            }
        }
    
    def clear_versions(self) -> None:
        """清空版本列表"""
        self._versions.clear()
    
    def _build_continuation_prompt(
        self, starting_text: str, word_count: int, direction: str
    ) -> str:
        """构建续写Prompt"""
        direction_hints = {
            ContinuationDirection.NATURAL: "按照故事自然发展规律，平滑推进情节",
            ContinuationDirection.PLOT_ADVANCE: "加速情节发展，推进主线",
            ContinuationDirection.CLIMAX_SETUP: "为后续高潮做铺垫和准备",
            ContinuationDirection.CHARACTER_DEEP: "深入刻画人物内心世界",
            ContinuationDirection.DIALOGUE: "以人物对话为主推进情节",
            ContinuationDirection.SCENE: "侧重环境和场景的描绘",
            ContinuationDirection.SUSPENSE: "设置悬念，引发读者好奇",
            ContinuationDirection.CONFLICT: "加剧人物或情节的冲突",
            ContinuationDirection.EMOTION: "强化情感表达和氛围",
            ContinuationDirection.TWIST: "设计一个意外的转折"
        }
        
        hint = direction_hints.get(direction, "自然发展")
        
        return f"""你是一位专业的小说续写专家。请根据以下信息续写小说：

## 原文内容
{starting_text}

## 续写要求
- 续写字数：约{word_count}字
- 续写方向：{hint}

## 续写原则
1. 保持原文的叙事风格和语言风格
2. 人物行为符合其性格设定
3. 情节发展自然流畅

请直接输出续写内容，不需要额外说明。"""
    
    def _call_llm(self, prompt: str, temperature: float) -> str:
        """调用LLM"""
        if self._llm_client:
            return self._llm_client.call(prompt, temperature=temperature)
        return "LLM客户端未配置"
    
    def _calculate_score(self, version: ContinuationVersion) -> float:
        """计算版本评分"""
        # 简化评分逻辑
        base_score = 0.5
        
        # 字数评分
        word_score = min(1.0, version.word_count / 1000) * 0.3
        
        # 温度评分（中等温度最佳）
        temp_score = 1.0 - abs(version.temperature - 0.8) * 0.2
        
        return base_score + word_score + temp_score * 0.2
```

```python
# plugin.py

from core.plugin_interface import (
    GeneratorPlugin, PluginMetadata, PluginType, PluginContext
)
from core.models import GenerationRequest, GenerationResult
from typing import Dict, Any, List, Tuple, Optional
from pydantic import BaseModel, Field
from enum import Enum
import uuid

class GenerationType(str, Enum):
    WORLDVIEW = "worldview"
    OUTLINE = "outline"
    CHARACTER = "character"
    PLOT = "plot"
    ALL = "all"

class DetailedLevel(str, Enum):
    QUICK = "quick"
    STANDARD = "standard"
    DETAILED = "detailed"

class QuickCreationRequest(BaseModel):
    """快捷创作请求"""
    keywords: List[str]
    target: GenerationType
    genre: str = ""
    generation_type: DetailedLevel = DetailedLevel.STANDARD
    reference_text: str = ""
    chapter_count: int = 10
    character_count: int = 3
    target_words: int = 100000

class QuickCreationResult(BaseModel):
    """快捷创作结果"""
    success: bool
    worldview: str = ""
    outline: str = ""
    characters: List[Dict[str, Any]] = []
    plot: str = ""
    error: str = ""

class QuickCreationPlugin(GeneratorPlugin):
    """快捷创作生成器插件"""
    
    PLUGIN_ID = "quick-creator-v1"
    PLUGIN_VERSION = "1.3.0"
    
    # Token限制配置
    TOKEN_LIMITS = {
        DetailedLevel.QUICK: (2000, 3000),
        DetailedLevel.STANDARD: (3000, 5000),
        DetailedLevel.DETAILED: (5000, 8000)
    }
    
    def __init__(self):
        super().__init__()
        self._llm_client = None
        self._cache: Dict[str, Any] = {}
    
    @classmethod
    def get_metadata(cls) -> PluginMetadata:
        return PluginMetadata(
            id=cls.PLUGIN_ID,
            name="快捷创作生成器 V1",
            version=cls.PLUGIN_VERSION,
            description="快速生成世界观、大纲、人物、情节等设定",
            author="Agent Pro Team",
            plugin_type=PluginType.GENERATOR
        )
    
    def initialize(self, context: PluginContext) -> bool:
        if not super().initialize(context):
            return False
        
        self._llm_client = context.service_locator.get("llm_client")
        return True
    
    def generate(self, request: GenerationRequest) -> GenerationResult:
        """生成内容（通用接口）"""
        # 简化实现，实际应根据请求类型调用不同生成方法
        return GenerationResult(
            request_id=request.request_id,
            content="生成结果",
            word_count=1000,
            iteration_count=1,
            validation_scores=None
        )
    
    def generate_worldview(
        self,
        keywords: List[str],
        reference_text: str = "",
        genre: str = "",
        generation_type: DetailedLevel = DetailedLevel.STANDARD
    ) -> str:
        """生成世界观"""
        prompt = self._build_worldview_prompt(keywords, reference_text, genre)
        
        min_tokens, max_tokens = self.TOKEN_LIMITS[generation_type]
        
        return self._call_llm(prompt, max_tokens)
    
    def generate_outline(
        self,
        theme: str,
        worldview_summary: str = "",
        reference_text: str = "",
        chapter_count: int = 10,
        target_words: int = 100000,
        generation_type: DetailedLevel = DetailedLevel.STANDARD
    ) -> str:
        """生成大纲"""
        prompt = self._build_outline_prompt(
            theme, worldview_summary, reference_text, chapter_count, target_words
        )
        
        min_tokens, max_tokens = self.TOKEN_LIMITS[generation_type]
        
        return self._call_llm(prompt, max_tokens)
    
    def generate_character(
        self,
        character_name: str,
        role: str,
        worldview_summary: str = "",
        reference_text: str = "",
        generation_type: DetailedLevel = DetailedLevel.STANDARD
    ) -> Dict[str, Any]:
        """生成人物"""
        prompt = self._build_character_prompt(
            character_name, role, worldview_summary, reference_text
        )
        
        content = self._call_llm(prompt, 4000)
        
        return self._parse_character_result(content)
    
    def generate_plot(
        self,
        outline_summary: str,
        characters: List[str],
        plot_type: str,
        reference_text: str = "",
        generation_type: DetailedLevel = DetailedLevel.STANDARD
    ) -> str:
        """生成情节"""
        prompt = self._build_plot_prompt(
            outline_summary, characters, plot_type, reference_text
        )
        
        min_tokens, max_tokens = self.TOKEN_LIMITS[generation_type]
        
        return self._call_llm(prompt, max_tokens)
    
    def generate_all(self, request: QuickCreationRequest) -> QuickCreationResult:
        """一键生成全部设定"""
        result = QuickCreationResult(success=True)
        
        try:
            # 1. 生成世界观（其他设定依赖）
            result.worldview = self.generate_worldview(
                request.keywords,
                request.reference_text,
                request.genre,
                request.generation_type
            )
            
            # 2. 生成大纲（基于世界观）
            result.outline = self.generate_outline(
                "、".join(request.keywords),
                result.worldview,
                request.reference_text,
                request.chapter_count,
                request.target_words,
                request.generation_type
            )
            
            # 3. 生成人物（基于世界观）
            for i in range(request.character_count):
                role = "主角" if i == 0 else f"配角{i}"
                char = self.generate_character(
                    "",  # AI自动命名
                    role,
                    result.worldview,
                    request.reference_text,
                    request.generation_type
                )
                result.characters.append(char)
            
            # 4. 生成情节（基于以上所有）
            result.plot = self.generate_plot(
                result.outline[:500],  # 大纲摘要
                [c.get("name", "") for c in result.characters],
                "main",
                request.reference_text,
                request.generation_type
            )
            
        except Exception as e:
            result.success = False
            result.error = str(e)
        
        return result
    
    def validate_request(self, request: GenerationRequest) -> Tuple[bool, List[str]]:
        """验证请求"""
        errors = []
        
        if not request.title:
            errors.append("关键词不能为空")
        
        return len(errors) == 0, errors
    
    def get_generation_options(self) -> Dict[str, Any]:
        """获取生成选项"""
        return {
            "target": {
                "type": "string",
                "enum": ["worldview", "outline", "character", "plot", "all"],
                "default": "all"
            },
            "generation_type": {
                "type": "string",
                "enum": ["quick", "standard", "detailed"],
                "default": "standard"
            },
            "chapter_count": {
                "type": "integer",
                "default": 10,
                "min": 5,
                "max": 100
            },
            "character_count": {
                "type": "integer",
                "default": 3,
                "min": 1,
                "max": 10
            }
        }
    
    def _build_worldview_prompt(
        self, keywords: List[str], reference_text: str, genre: str
    ) -> str:
        """构建世界观Prompt"""
        ref_section = f"\n参考文本：\n{reference_text}" if reference_text else ""
        
        return f"""你是一位专业的小说世界观构建专家。请根据以下关键词生成完整的世界观设定：

关键词：{"、".join(keywords)}
题材：{genre or "未指定"}
{ref_section}

请按以下结构输出世界观：
1. 世界背景（历史、时代背景）
2. 力量体系（魔法/科技/武功等）
3. 势力分布（主要势力及其关系）
4. 地理环境（重要地点、区域）
5. 社会规则（等级、制度、禁忌）
6. 特殊设定（独特元素）"""
    
    def _build_outline_prompt(
        self, theme: str, worldview: str, reference: str, chapters: int, words: int
    ) -> str:
        """构建大纲Prompt"""
        return f"""你是一位专业的小说大纲策划师。请根据以下信息生成完整的小说大纲：

主题：{theme}
世界观：{worldview[:500]}
预计章节数：{chapters}
目标字数：{words}

请按以下结构输出大纲：
1. 故事核心（主题、主线）
2. 篇幅规划
3. 分章大纲（每章标题、主要内容）"""
    
    def _build_character_prompt(
        self, name: str, role: str, worldview: str, reference: str
    ) -> str:
        """构建人物Prompt"""
        name_hint = f"姓名：{name}" if name else "请为主角取一个合适的名字"
        
        return f"""你是一位专业的人物设定师。请生成以下人物设定：

{name_hint}
角色定位：{role}
世界观背景：{worldview[:300]}

请输出完整的人物设定，包括：
外貌、性格、背景故事、能力、目标、弱点、人物关系"""
    
    def _build_plot_prompt(
        self, outline: str, characters: List[str], plot_type: str, reference: str
    ) -> str:
        """构建情节Prompt"""
        return f"""你是一位专业的情节设计师。请生成关键情节节点：

大纲摘要：{outline}
参与人物：{", ".join(characters)}
情节类型：{plot_type}

请输出：
1. 开端钩子
2. 核心冲突
3. 关键转折
4. 高潮场景
5. 结局走向"""
    
    def _parse_character_result(self, content: str) -> Dict[str, Any]:
        """解析人物生成结果"""
        # 简化实现
        return {
            "name": "待解析",
            "raw_content": content
        }
    
    def _call_llm(self, prompt: str, max_tokens: int) -> str:
        """调用LLM"""
        if self._llm_client:
            return self._llm_client.call(prompt, max_tokens=max_tokens)
        return "LLM客户端未配置"
```
