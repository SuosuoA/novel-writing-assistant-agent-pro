# UI搭建说明文档

## 概述

Novel Writing Assistant-Agent Pro 的GUI界面采用现代科技玻璃态（Glass Morphism）设计风格，实现无边框透明窗口、毛玻璃效果、发光边框等现代UI特性。

**版本**: V2.3
**创建日期**: 2026-03-22
**更新日期**: 2026-03-22
**技术栈**: Python 3.12.x + Tkinter + sv_ttk + Windows Acrylic API

### 视觉特点

1. **无边框窗口**：完全无边框设计，自定义标题栏，移除系统白边框
2. **毛玻璃效果**：Windows Acrylic/Mica效果
3. **悬浮卡片**：内容以卡片形式悬浮展示
4. **发光边框**：选中态带颜色高亮
5. **深色主题**：科技感深色UI

### 交互原则

1. **响应式按钮**：防抖保护 + 异步执行 + 加载状态
2. **页面切换**：侧边栏导航，无横向标签页
3. **线程安全回调**：所有UI更新通过 `root.after(0, callback)` 调度
4. **玻璃态动画**：悬停颜色过渡、点击反馈

## 核心模块

### 1. GlassWindowManager - 玻璃窗口管理器

**功能**：
- 启用Windows Acrylic效果
- 设置无边框窗口（完全移除系统边框）
- 控制窗口透明度

**使用示例**：
```python
from gui_main import GlassWindowManager

# 初始化
glass_mgr = GlassWindowManager(root)

# 启用Acrylic效果
glass_mgr.enable_acrylic(color=0x80000000)  # ARGB格式

# 设置无边框（移除系统边框和标题栏）
glass_mgr.make_frameless()

# 设置透明度
glass_mgr.set_window_opacity(0.95)
```

**功能**：
- 定义所有颜色变量
- 配置ttk Style
- 统一UI风格

```python
from gui_main import GlassTheme

# 使用主题色
bg_color = GlassTheme.GLASS_BG

# 应用到Style
style = ttk.Style()
```

## 技术要点

### 玻璃效果实现

```python
# Windows Acrylic效果
class GlassWindowManager:
    def enable_acrylic(self, color: int = 0x99000000) -> bool:
        accent = AccentPolicy()
        accent.AccentState = self.ACCENT_ENABLE_ACRYLICBLURBEHIND
        accent.AccentFlags = 2
        accent.GradientColor = color
        
        data = WindowCompositionAttrData()
        data.Attribute = self.WCA_ACCENT_POLICY
        data.Data = ctypes.cast(ctypes.byref(accent), ctypes.c_void_p)
        data.SizeOfData = ctypes.sizeof(accent)
        
        return self.user32.SetWindowCompositionAttribute(self.hwnd, ctypes.byref(data))
    
    def make_frameless(self) -> None:
        """完全移除系统边框"""
        style = self.user32.GetWindowLongW(self.hwnd, self.GWL_STYLE)
        style &= ~(self.WS_CAPTION | self.WS_SYSMENU)
        style |= self.WS_THICKFRAME | self.WS_MINIMIZEBOX | self.WS_MAXIMIZEBOX
        self.user32.SetWindowLongW(self.hwnd, self.GWL_STYLE, style)
```

### 线程安全机制

```python
# 所有UI操作通过root.after调度
def _invoke_callback(self, callback: Callable, *args) -> None:
    if self._root is None:
        callback(*args)
    else:
        self._root.after(0, lambda: callback(*args))
```

### 页面切换机制

```python
def _switch_to_page(self, page_id: str) -> None:
    """切换到指定页面"""
    # 隐藏当前页面
    if self._current_page and self._current_page in self._pages:
        self._pages[self._current_page].pack_forget()
    
    # 创建或显示新页面
    if page_id not in self._pages:
        self._pages[page_id] = self._create_page(page_id)
    
    self._pages[page_id].pack(fill=tk.BOTH, expand=True, in_=self._content_frame)
    self._current_page = page_id
```

### 工作台延迟创建机制（V2.2新增）

```python
def _create_all_workbench_tabs(self):
    """预创建工作台标签页（延迟创建策略，仅初始化为None）"""
    # 所有标签页初始化为None，首次切换时才创建
    # 这样可以避免首次点击工作台时一次性创建所有标签页导致卡顿
    for tab_id in ["worldview", "characters", "outline", "style", 
                   "generation", "reverse", "quick", "continue"]:
        self._workbench_tabs[tab_id] = None

def _switch_workbench_tab(self, tab_id: str):
    """切换工作台标签页（延迟创建）"""
    # ... 隐藏其他标签页 ...
    
    # 延迟创建：首次切换时才创建标签页内容
    if self._workbench_tabs.get(tab_id) is None:
        create_method = {
            "worldview": self._create_worldview_content,
            # ... 其他方法 ...
        }.get(tab_id)
        if create_method:
            self._workbench_tabs[tab_id] = create_method()
    
    # 显示当前标签页
    if self._workbench_tabs.get(tab_id) is not None:
        self._workbench_tabs[tab_id].pack(fill=tk.BOTH, expand=True, in_=self._workbench_content_frame)
```

```python
    """玻璃态主题配置"""
    # 色彩体系

    PRIMARY_LIGHT = "#60CDFF"  # 主色浅
```

```python
class MainWindow:
    def __init__(self, root):
        # 页面缓存字典
        self._pages = {}
        self._current_page = None
        self._workbench_tabs = {}  # 工作台子标签页缓存

    def _switch_to_page(self, page_id: str) -> None:
        """切换到指定页面"""
        # 隐藏当前页面
        if self._current_page and self._current_page in self._pages:
            self._pages[self._current_page].pack_forget()

        # 创建或显示新页面
        if page_id not in self._pages:
            self._pages[page_id] = self._create_page(page_id)

        # pack页面到content_frame
        page = self._pages[page_id]
        page.pack(fill=tk.BOTH, expand=True, in_=self._content_frame)
        self._current_page = page_id

        # 等待页面pack完成后更新Canvas宽度
        self.root.after(100, self._update_canvas_widths)
```

```python
scrollable_frame = ttk.Frame(canvas, style="TFrame")

def on_mousewheel(event):

canvas.bind("<Leave>", lambda e: canvas.unbind_all("<MouseWheel>"))
```

```python
def _create_all_workbench_tabs(self):
    """预创建工作台标签页（延迟创建策略）"""
    # 所有标签页初始化为None，首次切换时才创建
    for tab_id in ["worldview", "characters", "outline", "style",
                   "generation", "reverse", "quick", "continue"]:
        self._workbench_tabs[tab_id] = None

def _switch_workbench_tab(self, tab_id: str):
    """切换工作台标签页"""
    # 隐藏其他标签页
    for tid, frame in self._workbench_tabs.items():
        if frame is not None:
            frame.pack_forget()

    # 延迟创建：首次切换时才创建标签页内容
    if self._workbench_tabs.get(tab_id) is None:
        create_method = {
            "worldview": self._create_worldview_content,
            "characters": self._create_characters_content,
            "outline": self._create_outline_content,
            "style": self._create_style_content,
            "generation": self._create_generation_content,
            "reverse": self._create_reverse_content,
            "quick": self._create_quick_content,
            "continue": self._create_continue_content,
        }.get(tab_id)
        if create_method:
            self._workbench_tabs[tab_id] = create_method()

    # 显示当前标签页
    if self._workbench_tabs.get(tab_id) is not None:
        self._workbench_tabs[tab_id].pack(fill=tk.BOTH, expand=True,
                                           in_=self._workbench_content_frame)
```

### 关键修复经验

#### 修复1：页面内容缺失问题
**问题**：热榜、创作进度、项目管理、插件管理、设置页面内容无法显示
**根因**：Canvas窗口宽度未正确更新，scrollable_frame的宽度使用`canvas.winfo_width()`，但Canvas尚未完成pack
**解决方案**：
```python
def _switch_to_page(self, page_id: str) -> None:
    # ... 切换逻辑 ...
    self.root.after(100, self._update_canvas_widths)  # 延迟更新

def _update_canvas_widths(self):
    """递归更新所有Canvas的窗口宽度"""
    for widget in self._content_frame.winfo_children():
        self._update_canvas_width_recursive(widget)

def _update_canvas_width_recursive(self, widget):
    if widget.winfo_class() == 'Canvas':
        for item_id in widget.find_withtag("all"):
            tags = widget.gettags(item_id)
            if "all" not in tags:  # 这是canvas window
                widget.itemconfig(item_id, width=widget.winfo_width())
    for child in widget.winfo_children():
        self._update_canvas_width_recursive(child)
```

#### 修复2：快捷创作双重滚动条
**问题**：快捷创作页面内容被挤压在小格子里，有两个滚动条
**根因**：快捷创作内部创建了Canvas+Scrollbar，但工作台已有滚动容器
**解决方案**：移除快捷创作内部Canvas，直接使用workbench内容区的滚动
```python
# 错误方式（双重滚动）
def _create_quick_content(self):
    scroll_frame = ttk.Frame(self._workbench_content_frame)
    canvas = tk.Canvas(scroll_frame)
    scrollbar = ttk.Scrollbar(scroll_frame)
    # ... 滚动逻辑 ...

# 正确方式（单层滚动）
def _create_quick_content(self):
    # 直接在workbench内容区创建frame，不需要额外的Canvas
    frame = ttk.Frame(self._workbench_content_frame, style="TFrame")
```

#### 修复3：frame重复pack导致布局混乱
**问题**：快捷创作页面frame被重复pack，导致布局异常
**解决方案**：移除错误的`frame.pack()`语句（frame已通过canvas_window添加到canvas）

#### 修复4：工作台首次点击卡顿
**问题**：首次点击工作台会卡一下
**根因**：`_create_all_workbench_tabs()`一次性创建所有8个标签页
**解决方案**：改为延迟创建机制，所有标签页初始化为None，首次切换时才创建

#### 修复5：滚轮事件全局干扰
**问题**：鼠标滚轮影响整个窗口，而不是只影响滚动区域
**解决方案**：使用Enter/Leave事件控制bind_all/unbind_all
```python
canvas.bind("<Enter>", lambda e: canvas.bind_all("<MouseWheel>", on_mousewheel))
canvas.bind("<Leave>", lambda e: canvas.unbind_all("<MouseWheel>"))
```

### 性能优化总结

1. **页面缓存机制**：首次创建后缓存到_pages字典，后续切换直接pack()
2. **延迟创建机制**：工作台子标签页首次切换时才创建，避免一次性创建所有内容
3. **防抖机制**：按钮点击500ms防抖，避免重复触发
4. **线程安全回调**：所有UI操作通过`root.after(0, callback)`调度
5. **分层加载**：核心服务同步加载，UI组件异步加载

```python
def _create_scrollable_frame(self, parent: tk.Widget, **pack_options) -> tuple:
    """
    创建可滚动Frame的完整模板
    返回：(scrollable_frame, canvas, scrollbar)
    """
    # 创建滚动容器
    canvas = tk.Canvas(parent, bg=GlassTheme.GLASS_BG, highlightthickness=0)
    scrollbar = ttk.Scrollbar(parent, orient="vertical", command=canvas.yview)
    scrollable_frame = ttk.Frame(canvas, style="TFrame")

    # 添加到canvas
    canvas_window = canvas.create_window((0, 0), window=scrollable_frame, anchor="nw")
    canvas.configure(yscrollcommand=scrollbar.set)

    # 更新滚动区域和宽度
    def on_frame_configure(event):
        canvas.configure(scrollregion=canvas.bbox("all"))
        width = canvas.winfo_width()
        if width > 1:
            canvas.itemconfig(canvas_window, width=width)

    # 鼠标滚轮事件（Enter/Leave控制范围）
    def on_mousewheel(event):
        canvas.yview_scroll(int(-1 * (event.delta / 120)), "units")

    # 绑定事件
    scrollable_frame.bind("<Configure>", on_frame_configure)
    canvas.bind("<Configure>", on_frame_configure)
    canvas.bind("<Enter>", lambda e: canvas.bind_all("<MouseWheel>", on_mousewheel))
    canvas.bind("<Leave>", lambda e: canvas.unbind_all("<MouseWheel>"))

    # Linux滚轮支持
    canvas.bind("<Button-4>", lambda e: canvas.yview_scroll(-1, "units"))
    canvas.bind("<Button-5>", lambda e: canvas.yview_scroll(1, "units"))

    # pack（按顺序）
    scrollbar.pack(side=tk.RIGHT, fill=tk.Y)
    canvas.pack(side=tk.LEFT, fill=tk.BOTH, expand=True)

    return scrollable_frame, canvas, scrollbar
```

```python
            self._workbench_content_frame
        )
        self._workbench_content = scrollable_frame

    def _create_all_workbench_tabs(self):
        """
        预创建工作台标签页（延迟创建策略）
        关键：所有标签页初始化为None，首次切换时才创建
        这样可以避免首次点击工作台时一次性创建所有标签页导致卡顿
        """

                       "generation", "reverse", "quick", "continue"]:
            self._workbench_tabs[tab_id] = None

        """

                frame.pack_forget()

        # 延迟创建：首次切换时才创建标签页内容
        if self._workbench_tabs.get(tab_id) is None:
            create_method = {
                "worldview": self._create_worldview_content,
                "characters": self._create_characters_content,
                "outline": self._create_outline_content,
                "style": self._create_style_content,
                "generation": self._create_generation_content,
                "reverse": self._create_reverse_content,
                "quick": self._create_quick_content,
                "continue": self._create_continue_content,
            }.get(tab_id)

            if create_method:
                self._workbench_tabs[tab_id] = create_method()

        # 显示当前标签页
        if self._workbench_tabs.get(tab_id) is not None:
            self._workbench_tabs[tab_id].pack(
```

```python
class MainWindow:
    def __init__(self, root):
        self.root = root
        # 页面缓存字典
        self._pages = {}
        self._current_page = None

        # 创建所有页面框架
        self._create_all_pages()

    def _create_all_pages(self):
        """预创建所有页面"""
        page_creators = {
            "hot_ranking": self._create_hot_ranking_page,
            "workbench": self._create_workbench_page,
            "progress": self._create_progress_page,
            "project": self._create_project_page,
            "plugins": self._create_plugins_page,
            "settings": self._create_settings_page,
        }

        for page_id, creator in page_creators.items():
            self._pages[page_id] = creator()

    def _switch_to_page(self, page_id: str) -> None:
        """
        切换到指定页面
        关键：页面切换后延迟更新Canvas宽度，解决内容缺失问题
        """
        # 隐藏当前页面
        if self._current_page and self._current_page in self._pages:
            self._pages[self._current_page].pack_forget()

        # 显示新页面
        if page_id in self._pages:
            page = self._pages[page_id]
            page.pack(fill=tk.BOTH, expand=True, in_=self._content_frame)
            self._current_page = page_id

            # 关键修复：等待页面pack完成后更新Canvas宽度
            # Canvas可能在pack时宽度为0，导致scrollable_frame不可见
            # 延迟100ms确保Canvas完成pack和显示
            self.root.after(100, self._update_canvas_widths)

    def _update_canvas_widths(self):
        """
        递归更新所有Canvas的窗口宽度
        解决：scrollable_frame在Canvas中的宽度不正确
        """
        for widget in self._content_frame.winfo_children():
            self._update_canvas_width_recursive(widget)

    def _update_canvas_width_recursive(self, widget):
        """
        递归遍历所有控件，更新Canvas窗口宽度
        关键：识别Canvas控件并更新其canvas_window的宽度
        """
        # 如果是Canvas控件
        if widget.winfo_class() == 'Canvas':
            # 获取Canvas中的所有窗口
            for item_id in widget.find_withtag("all"):
                tags = widget.gettags(item_id)
                # "all"标签是Canvas自动添加的，canvas_window没有"all"标签
                if "all" not in tags:
                    # 这是canvas window，更新其宽度
                    width = widget.winfo_width()
                    if width > 1:  # 确保有有效宽度
                        widget.itemconfig(item_id, width=width)

        # 递归处理子控件
        for child in widget.winfo_children():
            self._update_canvas_width_recursive(child)
```

### 案例6：响应式按钮（防抖+异步+状态管理）

```python
class ResponsiveButton:
    """
    响应式按钮：防抖保护 + 异步执行 + 自动恢复
    """
    def __init__(self, parent, text: str, command: Callable,
                 async_handler=None, debounce_ms: int = 500,
                 auto_disable: bool = True):
        self.parent = parent
        self.command = command
        self.async_handler = async_handler
        self.debounce_ms = debounce_ms
        self.auto_disable = auto_disable
        self.last_click_time = 0
        self.is_disabled = False

        # 创建按钮
        self.button = tk.Button(parent, text=text, bg=GlassTheme.PRIMARY,
                              fg="white", font=("Microsoft YaHei", 10),
                              relief=tk.FLAT, cursor="hand2",
                              command=self._on_click)
        self.button.pack(fill=tk.X)

    def _on_click(self):
        """按钮点击处理"""
        current_time = time.time() * 1000  # 转换为毫秒

        # 防抖检查
        if current_time - self.last_click_time < self.debounce_ms:
            return

        self.last_click_time = current_time

        # 自动禁用
        if self.auto_disable:
            self.is_disabled = True
            original_text = self.button.cget("text")
            self.button.config(state=tk.DISABLED, text="处理中...")

        # 执行命令
        if self.async_handler:
            # 异步执行
            self.async_handler.execute(
                self.command,
                on_complete=lambda: self._restore_button(original_text)
            )
        else:
            # 同步执行
            self.command()
            if self.auto_disable:
                self._restore_button(original_text)

    def _restore_button(self, original_text: str):
        """恢复按钮状态"""
        self.is_disabled = False
        self.button.config(state=tk.NORMAL, text=original_text)
```

```python
            ]

        accent = AccentPolicy()
        accent.AccentState = self.ACCENT_ENABLE_ACRYLICBLURBEHIND
        accent.AccentFlags = 2
        accent.GradientColor = color

        data = WindowCompositionAttrData()
        data.Attribute = self.WCA_ACCENT_POLICY
        data.Data = ctypes.cast(ctypes.byref(accent), ctypes.c_void_p)
        data.SizeOfData = ctypes.sizeof(accent)

        return self.user32.SetWindowCompositionAttribute(
            self.hwnd, ctypes.byref(data)
        )

    def make_frameless(self, keep_resize_border: bool = False) -> None:
        """移除窗口边框"""
        WS_CAPTION = 0x00C00000
        WS_THICKFRAME = 0x00040000
        WS_SYSMENU = 0x00080000
        WS_MINIMIZEBOX = 0x00020000
        WS_MAXIMIZEBOX = 0x00010000
        GWL_STYLE = -16

        style = self.user32.GetWindowLongW(self.hwnd, GWL_STYLE)
        style &= ~(WS_CAPTION | WS_SYSMENU)
        style |= WS_THICKFRAME | WS_MINIMIZEBOX | WS_MAXIMIZEBOX
        self.user32.SetWindowLongW(self.hwnd, GWL_STYLE, style)


def main():
    """主函数"""
    root = tk.Tk()
    root.title("Novel Writing Assistant-Agent Pro")
    root.geometry("1280x720")

    # 应用玻璃主题
    style = ttk.Style()
    GlassTheme.apply_to_style(style)

    # 启用玻璃效果和无边框
    glass_mgr = GlassWindowManager(root)
    glass_mgr.enable_acrylic(color=0x99000000)
    glass_mgr.make_frameless(keep_resize_border=False)

    # 创建主窗口
    app = MainWindow(root)
    root.mainloop()


if __name__ == "__main__":
    main()
```
