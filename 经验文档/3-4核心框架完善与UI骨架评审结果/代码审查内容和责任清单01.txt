```python
def close_all(self) -> None:
    """关闭所有连接"""
    with self._connections_lock:
        # 标记所有连接为已关闭
        for conn in self._all_connections:
            try:
                conn.close()
            except Exception as e:
                logger.warning(f"Error closing connection: {e}")
        
        self._all_connections.clear()
    
    # 清理线程本地连接引用
    with self._lock:
        self._thread_local.connection = None
    
    logger.info("All database connections closed")
```

### P2-6: ConfigManager验证器增强

**问题描述**:

- 建议添加内置验证器（如类型检查）

```python
# 在ConfigValidator类中添加内置验证器
class ConfigValidator:
    def __init__(self):
        self._validators: Dict[str, Callable[[Any], bool]] = {}
        
        # 新增：内置类型验证器
        self._type_validators = {
            "int": lambda v: isinstance(v, int),
            "float": lambda v: isinstance(v, (int, float)),
            "str": lambda v: isinstance(v, str),
            "bool": lambda v: isinstance(v, bool),
            "list": lambda v: isinstance(v, list),
            "dict": lambda v: isinstance(v, dict),
        }
    
    def register_type_validator(self, key_path: str, type_name: str) -> None:
        """注册类型验证器"""
        if type_name in self._type_validators:
            self.register(key_path, self._type_validators[type_name])
```
