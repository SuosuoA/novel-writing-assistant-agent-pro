## 三、核心模块详解

### 3.1 EventBus

#### 3.1.1 设计理念

EventBus采用发布-订阅模式,实现模块间松耦合通信。

**核心概念**:
- Event:事件对象
- Publisher:事件发布者
- Subscriber:事件订阅者
- EventBus:事件总线

#### 3.1.2 使用示例

**定义事件**:
```python
from core.models import BaseEvent
from typing import Any, Dict
from datetime import datetime

class ChapterGeneratedEvent(BaseEvent):
    """章节生成事件"""
    
    event_type: str = "chapter.generated"
    chapter_num: int
    content: str
    score: float
    timestamp: datetime = datetime.now()
```

**发布事件**:
```python
from core.event_bus import EventBus

event_bus = EventBus()

# 发布事件
event = ChapterGeneratedEvent(
    chapter_num=10,
    content="章节内容...",
    score=0.85
)

event_bus.publish(event)
```

**订阅事件**:
```python
# 订阅事件
@event_bus.subscribe("chapter.generated")
def on_chapter_generated(event: ChapterGeneratedEvent):
    print(f"章节{event.chapter_num}已生成,评分:{event.score}")
```

#### 3.1.3 实现细节

```python
from typing import Callable, Dict, List, Any
from collections import defaultdict
import asyncio
import logging

class EventBus:
    """事件总线"""
    
    def __init__(self):
        self._subscribers: Dict[str, List[Callable]] = defaultdict(list)
        self._logger = logging.getLogger(__name__)
    
    def subscribe(self, event_type: str) -> Callable:
        """订阅事件装饰器"""
        def decorator(func: Callable) -> Callable:
            self._subscribers[event_type].append(func)
            return func
        return decorator
    
    def unsubscribe(self, event_type: str, func: Callable):
        """取消订阅"""
        if func in self._subscribers[event_type]:
            self._subscribers[event_type].remove(func)
    
    def publish(self, event: Any):
        """发布事件(同步)"""
        event_type = event.event_type
        subscribers = self._subscribers.get(event_type, [])
        
        for subscriber in subscribers:
            try:
                subscriber(event)
            except Exception as e:
                self._logger.error(f"事件处理失败: {e}")
                # 发布失败事件
                self._publish_error(event, e)
    
    async def publish_async(self, event: Any):
        """发布事件(异步)"""
        event_type = event.event_type
        subscribers = self._subscribers.get(event_type, [])
        
        tasks = []
        for subscriber in subscribers:
            if asyncio.iscoroutinefunction(subscriber):
                tasks.append(subscriber(event))
            else:
                tasks.append(asyncio.to_thread(subscriber, event))
        
        results = await asyncio.gather(*tasks, return_exceptions=True)
        
        for result in results:
            if isinstance(result, Exception):
                self._logger.error(f"事件处理失败: {result}")
                self._publish_error(event, result)
    
    def _publish_error(self, event: Any, error: Exception):
        """发布错误事件"""
        error_event = ErrorEvent(
            original_event=event,
            error=error
        )
        # 避免无限循环,直接调用错误处理器
        for handler in self._subscribers.get("error.occurred", []):
            try:
                handler(error_event)
            except Exception:
                pass  # 忽略错误处理器的错误
```

### 3.2 PluginRegistry

#### 3.2.1 插件生命周期

```
UNLOADED → LOADING → LOADED → READY → DISABLED
              ↓         ↓        ↓
            ERROR    ERROR    ERROR
```

#### 3.2.2 使用示例

**注册插件**:
```python
from core.plugin_registry import PluginRegistry
from core.plugin_interface import BasePlugin

registry = PluginRegistry()

# 创建插件实例
plugin = MyPlugin(config={"param": "value"})

# 注册插件
registry.register("my-plugin-v1", plugin)
```

**管理插件生命周期**:
```python
# 加载插件
registry.load("my-plugin-v1")

# 激活插件
registry.activate("my-plugin-v1")

# 停用插件
registry.deactivate("my-plugin-v1")

# 卸载插件
registry.unload("my-plugin-v1")
```

**查询插件状态**:
```python
# 获取插件
plugin = registry.get("my-plugin-v1")

# 获取状态
state = registry.get_state("my-plugin-v1")

# 获取所有插件
plugins = registry.get_all()

# 获取就绪插件
ready_plugins = registry.get_ready()
```

#### 3.2.3 实现细节

```python
from typing import Dict, Optional
from core.plugin_interface import BasePlugin, PluginState
import logging
import threading

class PluginRegistry:
    """插件注册表"""
    
    def __init__(self):
        self._plugins: Dict[str, BasePlugin] = {}
        self._states: Dict[str, PluginState] = {}
        self._lock = threading.RLock()
        self._logger = logging.getLogger(__name__)
    
    def register(self, plugin_id: str, plugin: BasePlugin):
        """注册插件"""
        with self._lock:
            if plugin_id in self._plugins:
                raise ValueError(f"插件已存在: {plugin_id}")
            
            self._plugins[plugin_id] = plugin
            self._states[plugin_id] = PluginState.UNLOADED
            self._logger.info(f"插件已注册: {plugin_id}")
    
    def unregister(self, plugin_id: str):
        """注销插件"""
        with self._lock:
            if plugin_id not in self._plugins:
                raise ValueError(f"插件不存在: {plugin_id}")
            
            # 确保插件已卸载
            if self._states[plugin_id] != PluginState.UNLOADED:
                self.unload(plugin_id)
            
            del self._plugins[plugin_id]
            del self._states[plugin_id]
            self._logger.info(f"插件已注销: {plugin_id}")
    
    def load(self, plugin_id: str, context: 'PluginContext') -> bool:
        """加载插件"""
        with self._lock:
            plugin = self._get_plugin(plugin_id)
            
            if self._states[plugin_id] != PluginState.UNLOADED:
                self._logger.warning(f"插件状态不正确: {plugin_id}")
                return False
            
            self._states[plugin_id] = PluginState.LOADING
            
            try:
                success = plugin.initialize(context)
                if success:
                    self._states[plugin_id] = PluginState.LOADED
                    self._logger.info(f"插件已加载: {plugin_id}")
                    return True
                else:
                    self._states[plugin_id] = PluginState.ERROR
                    self._logger.error(f"插件加载失败: {plugin_id}")
                    return False
            except Exception as e:
                self._states[plugin_id] = PluginState.ERROR
                self._logger.error(f"插件加载异常: {plugin_id}, {e}")
                return False
    
    def activate(self, plugin_id: str) -> bool:
        """激活插件"""
        with self._lock:
            plugin = self._get_plugin(plugin_id)
            
            if self._states[plugin_id] != PluginState.LOADED:
                self._logger.warning(f"插件状态不正确: {plugin_id}")
                return False
            
            try:
                success = plugin.activate()
                if success:
                    self._states[plugin_id] = PluginState.READY
                    self._logger.info(f"插件已激活: {plugin_id}")
                    return True
                else:
                    self._states[plugin_id] = PluginState.ERROR
                    self._logger.error(f"插件激活失败: {plugin_id}")
                    return False
            except Exception as e:
                self._states[plugin_id] = PluginState.ERROR
                self._logger.error(f"插件激活异常: {plugin_id}, {e}")
                return False
    
    def deactivate(self, plugin_id: str) -> bool:
        """停用插件"""
        with self._lock:
            plugin = self._get_plugin(plugin_id)
            
            if self._states[plugin_id] != PluginState.READY:
                self._logger.warning(f"插件状态不正确: {plugin_id}")
                return False
            
            try:
                success = plugin.deactivate()
                if success:
                    self._states[plugin_id] = PluginState.LOADED
                    self._logger.info(f"插件已停用: {plugin_id}")
                    return True
                else:
                    self._logger.error(f"插件停用失败: {plugin_id}")
                    return False
            except Exception as e:
                self._logger.error(f"插件停用异常: {plugin_id}, {e}")
                return False
    
    def unload(self, plugin_id: str) -> bool:
        """卸载插件"""
        with self._lock:
            plugin = self._get_plugin(plugin_id)
            
            # 先停用
            if self._states[plugin_id] == PluginState.READY:
                if not self.deactivate(plugin_id):
                    return False
            
            try:
                success = plugin.cleanup()
                if success:
                    self._states[plugin_id] = PluginState.UNLOADED
                    self._logger.info(f"插件已卸载: {plugin_id}")
                    return True
                else:
                    self._logger.error(f"插件卸载失败: {plugin_id}")
                    return False
            except Exception as e:
                self._logger.error(f"插件卸载异常: {plugin_id}, {e}")
                return False
    
    def get_state(self, plugin_id: str) -> Optional[PluginState]:
        """获取插件状态"""
        return self._states.get(plugin_id)
    
    def get(self, plugin_id: str) -> Optional[BasePlugin]:
        """获取插件"""
        return self._plugins.get(plugin_id)
    
    def get_all(self) -> Dict[str, BasePlugin]:
        """获取所有插件"""
        return self._plugins.copy()
    
    def get_ready(self) -> Dict[str, BasePlugin]:
        """获取就绪插件"""
        return {
            plugin_id: plugin
            for plugin_id, plugin in self._plugins.items()
            if self._states[plugin_id] == PluginState.READY
        }
    
    def _get_plugin(self, plugin_id: str) -> BasePlugin:
        """获取插件(内部方法)"""
        if plugin_id not in self._plugins:
            raise ValueError(f"插件不存在: {plugin_id}")
        return self._plugins[plugin_id]
```

### 3.3 ServiceLocator

#### 3.3.1 设计理念

ServiceLocator实现依赖注入模式,提供全局服务访问点。

**核心概念**:
- Service:服务对象
- ServiceLocator:服务定位器
- Scope:服务作用域

#### 3.3.2 使用示例

**注册服务**:
```python
from core.service_locator import ServiceLocator

# 注册单例服务
ServiceLocator.register_singleton("event_bus", event_bus)

# 注册工厂服务
ServiceLocator.register_factory("llm_client", lambda: create_llm_client())

#注册Agent
ServiceLocator.register_agent("thinker", thinker_agent)
```

**获取服务**:
```python
# 获取服务
event_bus = ServiceLocator.get("event_bus")
llm_client = ServiceLocator.get("llm_client")

# 获取Agent
thinker = ServiceLocator.get_agent("thinker")
```

#### 3.3.3 实现细节

```python
from typing import Any, Callable, Dict, Optional
import logging
import threading

class ServiceLocator:
    """服务定位器"""
    
    _instance = None
    _lock = threading.Lock()
    
    def __new__(cls):
        if cls._instance is None:
            with cls._lock:
                if cls._instance is None:
                    cls._instance = super().__new__(cls)
                    cls._instance._services = {}
                    cls._instance._factories = {}
                    cls._instance._agents = {}
                    cls._instance._logger = logging.getLogger(__name__)
        return cls._instance
    
    @classmethod
    def register_singleton(cls, name: str, instance: Any):
        """注册单例服务"""
        instance = cls()
        with cls._lock:
            if name in instance._services:
                raise ValueError(f"服务已存在: {name}")
            instance._services[name] = instance
            instance._logger.info(f"服务已注册: {name}")
    
    @classmethod
    def register_factory(cls, name: str, factory: Callable[[], Any]):
        """注册工厂服务"""
        instance = cls()
        with cls._lock:
            if name in instance._factories:
                raise ValueError(f"工厂已存在: {name}")
            instance._factories[name] = factory
            instance._logger.info(f"工厂已注册: {name}")
    
    @classmethod
    def register_agent(cls, name: str, agent: Any):
        """注册Agent"""
        instance = cls()
        with cls._lock:
            if name in instance._agents:
                raise ValueError(f"Agent已存在: {name}")
            instance._agents[name] = agent
            instance._logger.info(f"Agent已注册: {name}")
    
    @classmethod
    def get(cls, name: str) -> Any:
        """获取服务"""
        instance = cls()
        
        # 先查找单例
        if name in instance._services:
            return instance._services[name]
        
        # 再查找工厂
        if name in instance._factories:
            return instance._factories[name]()
        
        raise ValueError(f"服务不存在: {name}")
    
    @classmethod
    def get_agent(cls, name: str) -> Any:
        """获取Agent"""
        instance = cls()
        
        if name not in instance._agents:
            raise ValueError(f"Agent不存在: {name}")
        
        return instance._agents[name]
    
    @classmethod
    def unregister(cls, name: str):
        """注销服务"""
        instance = cls()
        with cls._lock:
            if name in instance._services:
                del instance._services[name]
            elif name in instance._factories:
                del instance._factories[name]
            elif name in instance._agents:
                del instance._agents[name]
            else:
                raise ValueError(f"服务不存在: {name}")
```

```python
        # 获取参数
        outline = params["outline"]
        characters = params["characters"]
        style = params["style"]

        for i in range(self.config["max_iterations"]):

                    "score": score,

            "score": score,
            "iterations": self.config["max_iterations"]
        }
```

## 五、Agent开发

### 5.1 Agent基类

```python
from abc import ABC, abstractmethod
from typing import Any, Dict, List

class BaseAgent(ABC):
    """Agent基类"""
    
    def __init__(self, name: str, config: Dict[str, Any]):
        self.name = name
        self.config = config
        self.logger = logging.getLogger(f"agents.{name}")
    
    @abstractmethod
    def execute(self, task: Dict[str, Any]) -> Dict[str, Any]:
        """执行任务"""
        pass
    
    def validate_task(self, task: Dict[str, Any]) -> bool:
        """验证任务"""
        required = self.get_required_fields()
        return all(field in task for field in required)
    
    @abstractmethod
    def get_required_fields(self) -> List[str]:
        """获取必需字段"""
        pass
    
    def log_execution(self, task: Dict[str, Any], result: Dict[str, Any]):
        """记录执行日志"""
        self.logger.info(f"Agent {self.name} executed task")
        self.logger.debug(f"Task: {task}")
        self.logger.debug(f"Result: {result}")
```
